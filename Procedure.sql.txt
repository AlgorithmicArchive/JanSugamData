CREATE PROCEDURE [dbo].[ChangePassword]
    @UserId INT,
    @CurrentPassword NVARCHAR(100),
    @NewPassword NVARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;

    -- Retrieve the stored hashed password for the user
    DECLARE @StoredHash VARBINARY(64);
    SELECT @StoredHash = [Password] FROM Users WHERE UserId = @UserId;

    -- Check if user exists
    IF @StoredHash IS NULL
    BEGIN
        SELECT 0 AS IsValid, 'User not found.' AS ErrorMessage;
        RETURN;
    END

    -- Hash the provided current password and compare
    DECLARE @InputHash VARBINARY(64);
    SET @InputHash = HASHBYTES('SHA2_256', @CurrentPassword);

    IF @InputHash != @StoredHash
    BEGIN
        SELECT 0 AS IsValid, 'Current password is incorrect.' AS ErrorMessage;
        RETURN;
    END

    -- Check new password length (additional backend validation)
    IF LEN(@NewPassword) < 8
    BEGIN
        SELECT 0 AS IsValid, 'New password must be at least 8 characters long.' AS ErrorMessage;
        RETURN;
    END

    -- Hash the new password
    DECLARE @NewHash VARBINARY(64);
    SET @NewHash = HASHBYTES('SHA2_256', @NewPassword);

    -- Update the password
    UPDATE Users
    SET [Password] = @NewHash
    WHERE UserId = @UserId;

    -- Return success
    SELECT 1 AS IsValid, '' AS ErrorMessage;
END;
GO

CREATE PROCEDURE dbo.GenerateDynamicReport
    @Tables NVARCHAR(MAX), -- e.g. 'Orders, Customers'
    @Columns NVARCHAR(MAX), -- e.g. 'Orders.OrderId, Customers.Name, JSON_VALUE(Orders.OrderDetails, ''$.ProductId'') AS ProductId'
    @Joins NVARCHAR(MAX) = NULL, -- e.g. 'INNER JOIN Customers ON Orders.CustomerId = Customers.CustomerId'
    @Filter NVARCHAR(MAX) = NULL, -- e.g. 'Orders.OrderDate >= ''2024-01-01'''
    @OrderBy NVARCHAR(MAX) = NULL -- e.g. 'Orders.OrderDate DESC'
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @SQL NVARCHAR(MAX);
    SET @SQL = N'SELECT ' + @Columns + N' FROM ' + @Tables + N' ';
    IF @Joins IS NOT NULL
        SET @SQL += ' ' + @Joins + ' ';
    IF @Filter IS NOT NULL
        SET @SQL += ' WHERE ' + @Filter + ' ';
    IF @OrderBy IS NOT NULL
        SET @SQL += ' ORDER BY ' + @OrderBy + ' ';
    PRINT @SQL; -- for debugging
    EXEC sp_executesql @SQL;
END;
GO

CREATE PROCEDURE [dbo].[GetAadhaarValidationCount]
    @AccessLevel VARCHAR(20), -- 'State' | 'District' | 'Tehsil' | 'Division'
    @AccessCode INT = NULL, -- used for District/Tehsil checks
    @ServiceId INT,
    @DivisionCode INT = NULL, -- Division filter
    @AadhaarFilter VARCHAR(20) = NULL -- optional: 'empty' | 'not_empty' | NULL
AS
BEGIN
    SET NOCOUNT ON;

    ---------------------------------------------------------------------
    -- Step 1: Pick only Citizen_Applications rows where DataType = 'legacy'
    ---------------------------------------------------------------------
    WITH RankedStatus AS (
        SELECT
            ca.ReferenceNumber,
            ca.ReferenceNumberAlphaNumeric,
            ca.ServiceId,
            ca.FormDetails,
            ca.[Status] AS workflow_status,
            ROW_NUMBER() OVER (
                PARTITION BY ca.ReferenceNumber
                ORDER BY COALESCE(jsonWorkFlow.completedAt, '9999-12-31') DESC, jsonWorkFlow.seq DESC
            ) AS rn
        FROM dbo.Citizen_Applications ca
        CROSS APPLY OPENJSON(ca.WorkFlow) WITH (
            status NVARCHAR(50) '$.status',
            completedAt DATETIME '$.completedAt',
            seq INT '$.seq'
        ) AS jsonWorkFlow
        WHERE ISJSON(ca.WorkFlow) = 1
          AND jsonWorkFlow.status IS NOT NULL
          AND jsonWorkFlow.status <> ''
          AND ca.ServiceId = @ServiceId
          AND ca.DataType = 'legacy'
    ),
    LatestStatus AS (
        SELECT
            ReferenceNumber,
            ReferenceNumberAlphaNumeric,
            ServiceId,
            FormDetails,
            workflow_status
        FROM RankedStatus
        WHERE rn = 1
    ),
    LocationData AS (
        SELECT
            ReferenceNumber,
            name,
            value,
            ROW_NUMBER() OVER (
                PARTITION BY ReferenceNumber
                ORDER BY
                    CASE
                        WHEN @AccessLevel = 'Tehsil' THEN
                            CASE WHEN name = 'Tehsil' THEN 1 ELSE 2 END
                        ELSE
                            CASE WHEN name = 'District' THEN 1 ELSE 2 END
                    END
            ) AS rn
        FROM LatestStatus ls
        OUTER APPLY OPENJSON(COALESCE(ls.FormDetails, '{}'), '$.Location')
            WITH (name NVARCHAR(50) '$.name', value INT '$.value') AS jl
        WHERE jl.name IN ('District', 'Tehsil')
    ),
    Filtered AS (
        SELECT DISTINCT
            ls.ReferenceNumber,
            ls.ReferenceNumberAlphaNumeric,
            ls.ServiceId,
            ls.FormDetails,
            ls.workflow_status,
            -- Aadhaar present flag
            CASE
                WHEN EXISTS (
                    SELECT 1
                    FROM OPENJSON(COALESCE(ls.FormDetails, '{}'), '$."Applicant Details"')
                         WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a
                    WHERE LOWER(LTRIM(RTRIM(ISNULL(a.name, '')))) = 'aadhaarnumber'
                      AND LTRIM(RTRIM(ISNULL(a.value, ''))) <> ''
                ) THEN 1 ELSE 0
            END AS AadhaarPresent
        FROM LatestStatus ls
        LEFT JOIN LocationData ld
            ON ls.ReferenceNumber = ld.ReferenceNumber
            AND ld.rn = 1
        LEFT JOIN District d
            ON ld.name = 'District' AND ld.value = d.DistrictID
        LEFT JOIN TSWOTehsil t
            ON ld.name = 'Tehsil' AND ld.value = t.TehsilID
        LEFT JOIN District d2
            ON t.DistrictID = d2.DistrictID
        WHERE LOWER(ls.workflow_status) = 'sanctioned'
          AND (
                -- State: Include all sanctioned applications
                @AccessLevel = 'State'
                -- District: Match District ID with AccessCode
                OR (@AccessLevel = 'District'
                    AND ld.name = 'District'
         AND ld.value = @AccessCode)
                -- Tehsil: Match Tehsil ID with AccessCode
                OR (@AccessLevel = 'Tehsil'
                    AND ld.name = 'Tehsil'
                    AND ld.value = @AccessCode)
                -- Division: Match District or Tehsil linked to DivisionCode
                OR (@AccessLevel = 'Division'
                    AND (
                        (ld.name = 'District' AND d.Division = @DivisionCode)
                        OR (ld.name = 'Tehsil' AND d2.Division = @DivisionCode)
                    )
                )
                -- Include applications with no valid location data for non-Division levels
                OR (@AccessLevel != 'Division' AND ld.name IS NULL)
            )
          AND (
                @AadhaarFilter IS NULL
                OR (
                    @AadhaarFilter = 'empty'
                    AND NOT EXISTS (
                        SELECT 1
                        FROM OPENJSON(COALESCE(ls.FormDetails, '{}'), '$."Applicant Details"')
                             WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a2
                        WHERE LOWER(LTRIM(RTRIM(ISNULL(a2.name, '')))) = 'aadhaarnumber'
                          AND LTRIM(RTRIM(ISNULL(a2.value, ''))) <> ''
                    )
                )
                OR (
                    @AadhaarFilter = 'not_empty'
                    AND EXISTS (
                        SELECT 1
                        FROM OPENJSON(COALESCE(ls.FormDetails, '{}'), '$."Applicant Details"')
                             WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a3
                        WHERE LOWER(LTRIM(RTRIM(ISNULL(a3.name, '')))) = 'aadhaarnumber'
                          AND LTRIM(RTRIM(ISNULL(a3.value, ''))) <> ''
                    )
                )
            )
    )
    ---------------------------------------------------------------------
    -- Debugging Output (Remove after verification)
    ---------------------------------------------------------------------
    -- SELECT
    -- f.ReferenceNumber,
    -- f.ReferenceNumberAlphaNumeric,
    -- f.workflow_status,
    -- f.AadhaarPresent,
    -- ld.name AS LocationName,
    -- ld.value AS LocationValue,
    -- d.Division AS DistrictDivision,
    -- d2.Division AS TehsilDivision
    -- FROM Filtered f
    -- LEFT JOIN LocationData ld
    -- ON f.ReferenceNumber = ld.ReferenceNumber
    -- AND ld.rn = 1
    -- LEFT JOIN District d
    -- ON ld.name = 'District' AND ld.value = d.DistrictID
    -- LEFT JOIN TSWOTehsil t
    -- ON ld.name = 'Tehsil' AND ld.value = t.TehsilID
    -- LEFT JOIN District d2
    -- ON t.DistrictID = d2.DistrictID
    -- WHERE @AccessLevel = 'Division' AND @DivisionCode = 2;

    ---------------------------------------------------------------------
    -- Step 2: Final aggregate counts
    ---------------------------------------------------------------------
    SELECT
        COUNT(*) AS TotalSanctioned,
        ISNULL(SUM(AadhaarPresent), 0) AS AadhaarValidated,
        COUNT(*) - ISNULL(SUM(AadhaarPresent), 0) AS AadhaarNotValidated
    FROM Filtered;
END;
GO

CREATE PROCEDURE [dbo].[GetAadhaarValidationData]
    @AccessLevel VARCHAR(20) = NULL,
    @AccessCode INT = NULL,
    @ServiceId INT,
    @DivisionCode INT = NULL,
    @AadhaarFilter VARCHAR(20) = NULL,
    @PageIndex INT = 0,
    @PageSize INT = 10,
    @IsPaginated BIT = 1,
    @TotalRecords INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    -- ======== Calculate Total Records ========
    WITH FilteredApplications AS (
        SELECT DISTINCT
            ca.ReferenceNumber
        FROM [dbo].[Citizen_Applications] ca
        OUTER APPLY (
            SELECT name, value
            FROM OPENJSON(COALESCE(ca.FormDetails, '{}'), '$.Location')
                 WITH (name NVARCHAR(50) '$.name', value INT '$.value')
            WHERE name = CASE
                            WHEN @AccessLevel = 'Tehsil' THEN 'Tehsil'
                            WHEN @AccessLevel = 'District' THEN 'District'
                            ELSE name
                         END
        ) AS jl
        LEFT JOIN [dbo].[District] d
            ON jl.name = 'District' AND jl.value = d.DistrictID
        LEFT JOIN [dbo].[TSWOTehsil] t
            ON jl.name = 'Tehsil' AND jl.value = t.TehsilID
        LEFT JOIN [dbo].[District] d2
            ON t.DistrictID = d2.DistrictID
        WHERE
            ca.ServiceId = @ServiceId
            AND ca.DataType = 'legacy'
            AND ISJSON(ca.WorkFlow) = 1
            AND LOWER(ca.[Status]) = 'sanctioned'
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District'
                    AND jl.name = 'District'
                    AND jl.value = @AccessCode)
                OR (@AccessLevel = 'Tehsil'
                    AND jl.name = 'Tehsil'
                    AND jl.value = @AccessCode)
                OR (
                    @AccessLevel = 'Division'
                    AND (
                        (jl.name = 'District' AND d.Division = @DivisionCode)
                        OR (jl.name = 'Tehsil' AND d2.Division = @DivisionCode)
                    )
                )
                OR (@AccessLevel != 'Division' AND jl.name IS NULL)
            )
            AND (
                @AadhaarFilter IS NULL
                OR (
                    @AadhaarFilter = 'empty'
                    AND NOT EXISTS (
                        SELECT 1
                        FROM OPENJSON(COALESCE(ca.FormDetails, '{}'), '$."Applicant Details"')
                             WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a
                        WHERE LOWER(LTRIM(RTRIM(ISNULL(a.name, '')))) = 'aadhaarnumber'
                          AND LTRIM(RTRIM(ISNULL(a.value, ''))) <> ''
                    )
                )
                OR (
                    @AadhaarFilter = 'not_empty'
                    AND EXISTS (
                        SELECT 1
                        FROM OPENJSON(COALESCE(ca.FormDetails, '{}'), '$."Applicant Details"')
                             WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a
                        WHERE LOWER(LTRIM(RTRIM(ISNULL(a.name, '')))) = 'aadhaarnumber'
                          AND LTRIM(RTRIM(ISNULL(a.value, ''))) <> ''
                    )
                )
            )
    )
    SELECT @TotalRecords = COUNT(*) FROM FilteredApplications;

    -- ======== Fetch Paginated Data ========
    CREATE TABLE #RankedApplications (
        ReferenceNumber NVARCHAR(50),
        ReferenceNumberAlphaNumeric NVARCHAR(50),
        Citizen_id INT,
        ServiceId INT,
        DistrictUidForBank NVARCHAR(50),
        FormDetails NVARCHAR(MAX),
        WorkFlow NVARCHAR(MAX),
        AdditionalDetails NVARCHAR(MAX),
        CurrentPlayer INT,
        [Status] NVARCHAR(50),
        DataType NVARCHAR(50),
        Created_at NVARCHAR(100),
        WorkflowStatus NVARCHAR(50),
        rn BIGINT,
  AadhaarPresent BIT
    );

    -- Populate #RankedApplications with one row per ReferenceNumber
    INSERT INTO #RankedApplications
    SELECT
        ca.ReferenceNumber,
        ca.ReferenceNumberAlphaNumeric,
        ca.Citizen_id,
        ca.ServiceId,
        ca.DistrictUidForBank,
        ca.FormDetails,
        ca.WorkFlow,
        ca.AdditionalDetails,
        ca.CurrentPlayer,
        ca.[Status],
        ca.DataType,
        ca.Created_at,
        LOWER(ca.[Status]) AS WorkflowStatus,
        ROW_NUMBER() OVER (
            ORDER BY COALESCE(wf.completedAt, '9999-12-31') DESC, wf.seq DESC
        ) AS rn,
        CASE
            WHEN EXISTS (
                SELECT 1
                FROM OPENJSON(COALESCE(ca.FormDetails, '{}'), '$."Applicant Details"')
                     WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a
                WHERE LOWER(LTRIM(RTRIM(ISNULL(a.name, '')))) = 'aadhaarnumber'
                  AND LTRIM(RTRIM(ISNULL(a.value, ''))) <> ''
            ) THEN 1 ELSE 0
        END AS AadhaarPresent
    FROM [dbo].[Citizen_Applications] ca
    CROSS APPLY (
        SELECT TOP 1 status, completedAt, seq
        FROM OPENJSON(ca.WorkFlow) WITH (
            status NVARCHAR(50) '$.status',
            completedAt DATETIME '$.completedAt',
            seq INT '$.seq'
        )
        WHERE status IS NOT NULL AND status <> ''
        ORDER BY COALESCE(completedAt, '9999-12-31') DESC, seq DESC
    ) AS wf
    WHERE
        ca.ServiceId = @ServiceId
        AND ca.DataType = 'legacy'
        AND ISJSON(ca.WorkFlow) = 1
        AND LOWER(ca.[Status]) = 'sanctioned'
        AND (
            @AadhaarFilter IS NULL
            OR (
                @AadhaarFilter = 'empty'
                AND NOT EXISTS (
                    SELECT 1
                    FROM OPENJSON(COALESCE(ca.FormDetails, '{}'), '$."Applicant Details"')
                         WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a
                    WHERE LOWER(LTRIM(RTRIM(ISNULL(a.name, '')))) = 'aadhaarnumber'
                      AND LTRIM(RTRIM(ISNULL(a.value, ''))) <> ''
                )
            )
            OR (
                @AadhaarFilter = 'not_empty'
                AND EXISTS (
                    SELECT 1
                    FROM OPENJSON(COALESCE(ca.FormDetails, '{}'), '$."Applicant Details"')
                         WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a
                    WHERE LOWER(LTRIM(RTRIM(ISNULL(a.name, '')))) = 'aadhaarnumber'
                      AND LTRIM(RTRIM(ISNULL(a.value, ''))) <> ''
                )
            )
        );

    SELECT
        ra.ReferenceNumber,
        ra.ReferenceNumberAlphaNumeric,
        ra.Citizen_id,
        ra.ServiceId,
        ra.DistrictUidForBank,
        ra.FormDetails,
        ra.WorkFlow,
        ra.AdditionalDetails,
        ra.CurrentPlayer,
        ra.[Status],
        ra.DataType,
        ra.Created_at,
        ra.WorkflowStatus,
        ra.rn,
        ra.AadhaarPresent,
        d.DistrictName,
        t.TehsilName
    INTO #FilteredApplications
    FROM #RankedApplications ra
    OUTER APPLY (
        SELECT name, value
        FROM OPENJSON(COALESCE(ra.FormDetails, '{}'), '$.Location')
             WITH (name NVARCHAR(50) '$.name', value INT '$.value')
        WHERE name = CASE
                        WHEN @AccessLevel = 'Tehsil' THEN 'Tehsil'
                        WHEN @AccessLevel = 'District' THEN 'District'
                        ELSE name
                     END
    ) AS jl
    LEFT JOIN [dbo].[District] d
        ON jl.name = 'District' AND jl.value = d.DistrictID
    LEFT JOIN [dbo].[TSWOTehsil] t
        ON jl.name = 'Tehsil' AND jl.value = t.TehsilID
    LEFT JOIN [dbo].[District] d2
        ON t.DistrictID = d2.DistrictID
    WHERE
        (
            @AccessLevel = 'State'
            OR (@AccessLevel = 'District'
                AND jl.name = 'District'
                AND jl.value = @AccessCode)
            OR (@AccessLevel = 'Tehsil'
               AND jl.name = 'Tehsil'
                AND jl.value = @AccessCode)
            OR (
                @AccessLevel = 'Division'
                AND (
                    (jl.name = 'District' AND d.Division = @DivisionCode)
                    OR (jl.name = 'Tehsil' AND d2.Division = @DivisionCode)
                )
            )
            OR (@AccessLevel != 'Division' AND jl.name IS NULL)
        );

    SELECT
        fa.ReferenceNumber,
        fa.ReferenceNumberAlphaNumeric,
        fa.Citizen_id,
        fa.ServiceId,
        fa.DistrictUidForBank,
        fa.FormDetails,
        fa.WorkFlow,
        fa.AdditionalDetails,
        fa.CurrentPlayer,
        fa.[Status],
        fa.DataType,
        fa.Created_at,
        fa.WorkflowStatus,
        fa.AadhaarPresent,
        fa.DistrictName,
        fa.TehsilName
    FROM #FilteredApplications fa
    WHERE
        (@IsPaginated = 1 AND fa.rn BETWEEN (@PageIndex * @PageSize) + 1 AND ((@PageIndex + 1) * @PageSize))
        OR (@IsPaginated = 0)
    ORDER BY fa.rn;

    -- Cleanup
    DROP TABLE #RankedApplications;
    DROP TABLE #FilteredApplications;
END;
GO

CREATE PROCEDURE [dbo].[GetAgeAndPensionCounts]
    @ServiceId INT,
    @AccessLevel VARCHAR(20),
    @AccessCode INT,
    @ApplicationStatus VARCHAR(50) = NULL,
    @DataType VARCHAR(20) = NULL,
    @StartDate DATE = NULL,
    @EndDate DATE = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- Step 1: Extract DOB and PensionType and filter
    WITH Data_CTE AS
    (
        SELECT
            ca.ReferenceNumber,
            TRY_CAST(JSON_VALUE(ad.value, '$.value') AS DATE) AS DateOfBirth,
            JSON_VALUE(pt.value, '$.value') AS PensionType
        FROM dbo.Citizen_Applications ca
        OUTER APPLY OPENJSON(ca.FormDetails, '$."Applicant Details"') AS ad
        OUTER APPLY OPENJSON(ca.FormDetails, '$."Pension Type"') AS pt
        OUTER APPLY OPENJSON(ca.FormDetails, '$.Location')
            WITH (name NVARCHAR(50) '$.name', value INT '$.value') AS loc
        LEFT JOIN [dbo].[District] d
            ON loc.name = 'District' AND loc.value = d.DistrictID
        WHERE
            ca.ServiceId = @ServiceId
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District' AND loc.name = 'District' AND loc.value = @AccessCode)
                OR (@AccessLevel = 'Tehsil' AND loc.name = 'Tehsil' AND loc.value = @AccessCode)
                OR (@AccessLevel = 'Division' AND loc.name = 'District' AND d.Division = @AccessCode)
            )
            AND (
                @ApplicationStatus = 'total'
                OR (@ApplicationStatus IS NOT NULL AND ca.Status = @ApplicationStatus)
            )
            AND (@DataType IS NULL OR ca.DataType = @DataType)
            AND JSON_VALUE(ad.value, '$.name') = 'DateOfBirth'
            AND TRY_CAST(JSON_VALUE(ad.value, '$.value') AS DATE) IS NOT NULL
            AND (@StartDate IS NULL OR TRY_CAST(JSON_VALUE(ad.value, '$.value') AS DATE) >= @StartDate)
            AND (@EndDate IS NULL OR TRY_CAST(JSON_VALUE(ad.value, '$.value') AS DATE) <= @EndDate)
    ),
    Age_CTE AS
    (
        SELECT
            ReferenceNumber,
            PensionType,
            DATEDIFF(YEAR, DateOfBirth, GETDATE())
            - CASE
                WHEN MONTH(DateOfBirth) > MONTH(GETDATE())
                     OR (MONTH(DateOfBirth) = MONTH(GETDATE()) AND DAY(DateOfBirth) > DAY(GETDATE()))
                THEN 1 ELSE 0
              END AS Age
        FROM Data_CTE
    )

    -- Step 2: Return count by Age and PensionType
    SELECT
        Age,
        PensionType,
        COUNT(*) AS CountOfApplicants
    FROM Age_CTE
    GROUP BY Age, PensionType
    ORDER BY Age, PensionType;
END;
GO

CREATE PROCEDURE [dbo].[GetAgeCountsFiltered]
    @ServiceId INT,
    @AccessLevel VARCHAR(20),
    @AccessCode INT,
    @ApplicationStatus VARCHAR(50) = NULL,
    @DataType VARCHAR(20) = NULL,
    @StartDate DATE = NULL,
    @EndDate DATE = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- Step 1: Extract DOB and filter
    WITH DOB_CTE AS
    (
        SELECT
            ca.ReferenceNumber,
            TRY_CAST(JSON_VALUE(ad.value, '$.value') AS DATE) AS DateOfBirth
        FROM dbo.Citizen_Applications ca
        OUTER APPLY OPENJSON(ca.FormDetails, '$."Applicant Details"') AS ad
        OUTER APPLY OPENJSON(ca.FormDetails, '$.Location')
            WITH (name NVARCHAR(50) '$.name', value INT '$.value') AS loc
        LEFT JOIN [dbo].[District] d
            ON loc.name = 'District' AND loc.value = d.DistrictID
        WHERE
            ca.ServiceId = @ServiceId
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District' AND loc.name = 'District' AND loc.value = @AccessCode)
                OR (@AccessLevel = 'Tehsil' AND loc.name = 'Tehsil' AND loc.value = @AccessCode)
                OR (@AccessLevel = 'Division' AND loc.name = 'District' AND d.Division = @AccessCode)
            )
            AND (
                @ApplicationStatus = 'total'
                OR (@ApplicationStatus IS NOT NULL AND ca.Status = @ApplicationStatus)
            )
            AND (@DataType IS NULL OR ca.DataType = @DataType)
            AND JSON_VALUE(ad.value, '$.name') = 'DateOfBirth'
            AND TRY_CAST(JSON_VALUE(ad.value, '$.value') AS DATE) IS NOT NULL
            AND (@StartDate IS NULL OR TRY_CAST(JSON_VALUE(ad.value, '$.value') AS DATE) >= @StartDate)
            AND (@EndDate IS NULL OR TRY_CAST(JSON_VALUE(ad.value, '$.value') AS DATE) <= @EndDate)
    ),
    Age_CTE AS
    (
        SELECT
            ReferenceNumber,
            DATEDIFF(YEAR, DateOfBirth, GETDATE())
            - CASE
                WHEN MONTH(DateOfBirth) > MONTH(GETDATE())
                     OR (MONTH(DateOfBirth) = MONTH(GETDATE()) AND DAY(DateOfBirth) > DAY(GETDATE()))
                THEN 1 ELSE 0
              END AS Age
        FROM DOB_CTE
    )

    -- Step 2: Return count by distinct age
    SELECT
        Age,
        COUNT(*) AS CountOfApplicants
    FROM Age_CTE
    GROUP BY Age
    ORDER BY Age;
END;
GO

CREATE PROCEDURE [dbo].[GetAndIncrementCount]
    @DistrictId INT,
    @ServiceId INT,
    @FinancialYear NVARCHAR(20),
    @Type NVARCHAR(30),
    @NewCount INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    BEGIN TRANSACTION;

    BEGIN TRY
        -- Attempt to find the record
        IF EXISTS (SELECT 1 FROM ApplicationPerDistrict
                   WHERE DistrictId = @DistrictId AND ServiceId = @ServiceId AND [Type] = @Type)
        BEGIN
            -- Update existing and get new count
            UPDATE ApplicationPerDistrict
            SET CountValue = CountValue + 1
            WHERE DistrictId = @DistrictId AND ServiceId = @ServiceId AND [Type] = @Type ;

            SELECT @NewCount = CountValue
            FROM ApplicationPerDistrict
            WHERE DistrictId = @DistrictId AND ServiceId = @ServiceId AND [Type] = @Type;
        END
        ELSE
        BEGIN
            -- Insert new
            INSERT INTO ApplicationPerDistrict ([Type],DistrictId, ServiceId, FinancialYear, CountValue)
            VALUES (@Type,@DistrictId, @ServiceId, @FinancialYear, 1);

            SET @NewCount = 1;
        END

        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END;
GO

CREATE PROCEDURE [dbo].[GetApplicationForAadhaarValidation]
    @Role VARCHAR(255),
    @AccessLevel VARCHAR(20),
    @AccessCode INT,
    @ApplicationStatus VARCHAR(50) = NULL,
    @ServiceId INT,
    @PageIndex INT = 0,
    @PageSize INT = 10,
    @IsPaginated BIT = 1,
    @DataType VARCHAR(20) = NULL,
    @AadhaarFilter VARCHAR(20) = NULL, -- New parameter: 'empty', 'not_empty', or NULL
    @TotalRecords INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    ---------------------------------------------------------------------
    -- Step 0: Temp table to hold filtered results
    ---------------------------------------------------------------------
    IF OBJECT_ID('tempdb..#Filtered') IS NOT NULL
        DROP TABLE #Filtered;

    CREATE TABLE #Filtered
    (
        ReferenceNumber NVARCHAR(100) PRIMARY KEY,
        ReferenceNumberAlphaNumeric NVARCHAR(100),
        Citizen_id INT,
        ServiceId INT,
        DistrictUidForBank NVARCHAR(50),
        FormDetails NVARCHAR(MAX),
        WorkFlow NVARCHAR(MAX),
        AdditionalDetails NVARCHAR(MAX),
        CurrentPlayer INT,
        [Status] NVARCHAR(50),
        DataType NVARCHAR(20),
        Created_at NVARCHAR(100)
    );

    ---------------------------------------------------------------------
    -- Step 1: Insert filtered data into #Filtered
    -- Dynamic Aadhaar filter based on @AadhaarFilter
    ---------------------------------------------------------------------
    INSERT INTO #Filtered (ReferenceNumber, ReferenceNumberAlphaNumeric, Citizen_id, ServiceId, DistrictUidForBank,
                           FormDetails, WorkFlow, AdditionalDetails, CurrentPlayer, [Status], DataType, Created_at)
    SELECT DISTINCT
        ca.ReferenceNumber,
        ca.ReferenceNumberAlphaNumeric,
        ca.Citizen_id,
        ca.ServiceId,
        ca.DistrictUidForBank,
        ca.FormDetails,
        ca.WorkFlow,
        ca.AdditionalDetails,
        ca.CurrentPlayer,
        ca.[Status],
        ca.DataType,
        ca.Created_at
    FROM [dbo].[Citizen_Applications] ca
    CROSS APPLY OPENJSON(ca.WorkFlow) AS wf
    OUTER APPLY OPENJSON(ca.FormDetails, '$.Location')
        WITH (name NVARCHAR(200) '$.name', value NVARCHAR(200) '$.value') AS jsonLocation
    LEFT JOIN [dbo].[District] d
        ON jsonLocation.name = 'District' AND jsonLocation.value = d.DistrictID
    WHERE
        ca.ServiceId = @ServiceId
        AND (
            @DataType = 'legacy'
            OR JSON_VALUE(wf.value, '$.designation') = @Role
        )
        AND (
            @AccessLevel = 'State'
            OR (@AccessLevel = 'District' AND jsonLocation.name = 'District' AND jsonLocation.value = CAST(@AccessCode AS NVARCHAR(200)))
            OR (@AccessLevel = 'Tehsil' AND jsonLocation.name = 'Tehsil' AND jsonLocation.value = CAST(@AccessCode AS NVARCHAR(200)))
            OR (@AccessLevel = 'Division' AND jsonLocation.name = 'District' AND d.Division = @AccessCode)
        )
        AND ca.[Status] IN ('Sanctioned', 'sanctioned')
        AND (@DataType IS NULL OR ca.DataType = @DataType)
        -- Dynamic Aadhaar filter
        AND (
            @AadhaarFilter IS NULL -- No Aadhaar filter
            OR (
                @AadhaarFilter = 'empty'
                AND NOT EXISTS (
                    SELECT 1
                    FROM OPENJSON(ca.FormDetails, '$."Applicant Details"')
                         WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a
                    WHERE LOWER(a.name) = 'aadhaarnumber'
                      AND LTRIM(RTRIM(ISNULL(a.value, ''))) <> ''
                )
            )
            OR (
                @AadhaarFilter = 'not_empty'
                AND EXISTS (
                    SELECT 1
                    FROM OPENJSON(ca.FormDetails, '$."Applicant Details"')
                         WITH (name NVARCHAR(200) '$.name', value NVARCHAR(4000) '$.value') AS a
                    WHERE LOWER(a.name) = 'aadhaarnumber'
                      AND LTRIM(RTRIM(ISNULL(a.value, ''))) <> ''
                )
            )
        );

    ---------------------------------------------------------------------
    -- Step 2: Count total records
    ---------------------------------------------------------------------
    SELECT @TotalRecords = COUNT(*) FROM #Filtered;

    ---------------------------------------------------------------------
    -- Step 3: Paginate results
    ---------------------------------------------------------------------
    ;WITH Numbered AS (
        SELECT
            f.ReferenceNumber,
            f.ReferenceNumberAlphaNumeric,
            MAX(f.Citizen_id) AS Citizen_id,
            MAX(f.ServiceId) AS ServiceId,
            MAX(f.DistrictUidForBank) AS DistrictUidForBank,
            MAX(f.FormDetails) AS FormDetails,
            MAX(f.WorkFlow) AS WorkFlow,
            MAX(f.AdditionalDetails) AS AdditionalDetails,
            MAX(f.CurrentPlayer) AS CurrentPlayer,
            MAX(f.[Status]) AS [Status],
            MAX(f.DataType) AS DataType,
            MAX(f.Created_at) AS Created_at,
            ROW_NUMBER() OVER (
                ORDER BY TRY_CAST(PARSENAME(REPLACE(f.ReferenceNumber, '/', '.'), 1) AS INT)
            ) AS RowNum
        FROM #Filtered f
        GROUP BY f.ReferenceNumber, f.ReferenceNumberAlphaNumeric
    )
    SELECT
        n.ReferenceNumber,
        n.ReferenceNumberAlphaNumeric,
        n.Citizen_id,
        n.ServiceId,
        n.DistrictUidForBank,
        n.FormDetails,
        n.WorkFlow,
        n.AdditionalDetails,
        n.CurrentPlayer,
        n.[Status],
        n.DataType,
        n.Created_at
    FROM Numbered n
    WHERE
        (@IsPaginated = 1 AND n.RowNum BETWEEN (@PageIndex * @PageSize) + 1 AND ((@PageIndex + 1) * @PageSize))
        OR (@IsPaginated = 0)
    ORDER BY n.RowNum;

END;

CREATE PROCEDURE [dbo].[GetApplicationForCorrigendum]
    @ReferenceNumber VARCHAR(50),
    @Role VARCHAR(255),
    @OfficerAccessLevel VARCHAR(50),
    @OfficerAccessCode INT,
    @ServiceId INT,
    @Type VARCHAR(50),
    @Message NVARCHAR(255) OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @IsSanctioned BIT = 0;
    DECLARE @IsAccessible BIT = 0;
    DECLARE @IsAuthorized BIT = 0;
    DECLARE @CurrentPlayerStr NVARCHAR(10);

    -- 1. Check if application is sanctioned
    IF EXISTS (
        SELECT 1 FROM [dbo].[Citizen_Applications]
        WHERE ReferenceNumber = @ReferenceNumber AND Status = 'Sanctioned'
    )
        SET @IsSanctioned = 1;

    -- 2. Validate type vs sanction status
    IF @Type = 'Correction' AND @IsSanctioned = 1
    BEGIN
        SET @Message = 'Error: Correction is not allowed for sanctioned applications.';
        SELECT * FROM (SELECT 1 AS x) AS t WHERE 1=0; -- Return empty result
        RETURN;
    END
    ELSE IF @Type IN ('Corrigendum', 'Amendment','Withheld') AND @IsSanctioned = 0
    BEGIN
        SET @Message = 'Error: ' + @Type + ' can only be issued for sanctioned applications.';
        SELECT * FROM (SELECT 1 AS x) AS t WHERE 1=0;
        RETURN;
    END
    ELSE IF @Type NOT IN ('Corrigendum', 'Correction', 'Amendment','Withheld')
    BEGIN
        SET @Message = 'Error: Invalid type. Must be Corrigendum, Correction, Amendment, or Withheld.';
        SELECT * FROM (SELECT 1 AS x) AS t WHERE 1=0;
        RETURN;
    END

    -- 3. Check officer access level (Location-based)
    IF EXISTS (
        SELECT 1
        FROM [dbo].[Citizen_Applications] ca
        OUTER APPLY (
            SELECT *
            FROM OPENJSON(ca.FormDetails, '$.Location')
            WITH (
                name NVARCHAR(50) '$.name',
                value INT '$.value'
            )
        ) AS jsonLocation
        LEFT JOIN [dbo].[District] d ON jsonLocation.name = 'District' AND jsonLocation.value = d.DistrictID
        WHERE ca.ReferenceNumber = @ReferenceNumber
          AND (
                @OfficerAccessLevel = 'State'
             OR (@OfficerAccessLevel = 'District' AND jsonLocation.name = 'District' AND jsonLocation.value = @OfficerAccessCode)
             OR (@OfficerAccessLevel = 'Tehsil' AND jsonLocation.name = 'Tehsil' AND jsonLocation.value = @OfficerAccessCode)
             OR (@OfficerAccessLevel = 'Division' AND jsonLocation.name = 'District' AND d.Division = @OfficerAccessCode)
          )
    )
        SET @IsAccessible = 1;
    ELSE
    BEGIN
        SET @Message = 'Error: Application does not belong to the specified access level and code.';
        SELECT * FROM (SELECT 1 AS x) AS t WHERE 1=0;
        RETURN;
    END

    -- 4. Authorization check - Replace JSON_PATH_EXISTS with ISJSON + JSON_VALUE
    SELECT @CurrentPlayerStr = LTRIM(RTRIM(STR(CurrentPlayer)))
    FROM [dbo].[Citizen_Applications]
    WHERE ReferenceNumber = @ReferenceNumber;

    IF @Type = 'Correction'
    BEGIN
        IF EXISTS (
            SELECT 1
            FROM [dbo].[Citizen_Applications] ca
            WHERE ca.ReferenceNumber = @ReferenceNumber
              AND ca.ServiceId = @ServiceId
              AND ISJSON(ca.WorkFlow) = 1
              AND TRY_CONVERT(INT, @CurrentPlayerStr) IS NOT NULL
              AND JSON_VALUE(ca.WorkFlow, '$[' + @CurrentPlayerStr + '].designation') = @Role
        )
            SET @IsAuthorized = 1;
        ELSE IF @Message IS NULL
            SET @Message = 'Error: You are not the Current Officer of this Application to perform a Correction.';
    END
    ELSE IF @Type IN ('Corrigendum', 'Amendment', 'Withheld')
        SET @IsAuthorized = 1;

    -- 5. Final success message
    IF @Message IS NULL
    BEGIN
        IF @Type IN ('Corrigendum', 'Amendment', 'Withheld')
            SET @Message = 'Success: ' + @Type + ' data retrieved.';
        ELSE IF @Type = 'Correction'
            SET @Message = 'Success: Correction data retrieved.';
    END

    -- 6. Final SELECT - Also fixed JSON_PATH_EXISTS here
    SELECT
        ca.ReferenceNumber,
        ca.ReferenceNumberAlphaNumeric,
        ca.Citizen_id,
        ca.ServiceId,
        ca.DistrictUidForBank,
        ca.FormDetails,
        ca.WorkFlow,
        ca.AdditionalDetails,
        ca.CurrentPlayer,
        ca.Status,
        ca.DataType,
        ca.Created_at,
        c.[type],
        c.CorrigendumFields AS Corrigendum,
        c.Type AS CorrigendumType
    FROM [dbo].[Citizen_Applications] ca
    LEFT JOIN [dbo].[Corrigendum] c
        ON ca.ReferenceNumber = c.ReferenceNumber AND c.Type = @Type
    WHERE ca.ServiceId = @ServiceId
      AND ca.ReferenceNumber = @ReferenceNumber
      AND (
            (@Type IN ('Corrigendum', 'Amendment','Withheld') AND ca.Status = 'Sanctioned')
         OR (@Type = 'Correction'
             AND ca.Status != 'Sanctioned'
             AND ISJSON(ca.WorkFlow) = 1
             AND TRY_CONVERT(INT, LTRIM(RTRIM(STR(ca.CurrentPlayer)))) IS NOT NULL
             AND JSON_VALUE(ca.WorkFlow, '$[' + LTRIM(RTRIM(STR(ca.CurrentPlayer))) + '].designation') = @Role)
      );
END;
GO

CREATE PROCEDURE [dbo].[GetApplicationsByAccessLevel]
    @AccessLevel VARCHAR(50), -- 'State', 'Division', or 'District'
    @AccessCode INT, -- 0 for State, or actual DivisionId/DistrictId
    @DataType NVARCHAR(50) = NULL, -- Optional filter
    @Department INT = NULL -- Optional department filter
AS
BEGIN
    SET NOCOUNT ON;

    -- Validate AccessLevel
    IF @AccessLevel NOT IN ('State', 'Division', 'District')
    BEGIN
        SELECT
            CAST(NULL AS NVARCHAR(50)) AS ReferenceNumber,
            CAST(NULL AS NVARCHAR(MAX)) AS FormDetails,
            CAST(NULL AS DATETIME) AS Created_at
        WHERE 1 = 0; -- Return empty result set
        RETURN;
    END;

    IF @AccessLevel = 'State'
    BEGIN
        SELECT
            ca.ReferenceNumber,
            ca.FormDetails,
            ca.Created_at
        FROM Citizen_Applications ca
        INNER JOIN Services s ON ca.ServiceId = s.ServiceId
        WHERE (@DataType IS NULL OR ca.DataType = @DataType)
          AND (@Department IS NULL OR s.DepartmentId = @Department);
    END
    ELSE IF @AccessLevel = 'Division'
    BEGIN
        SELECT
            ca.ReferenceNumber,
            ca.FormDetails,
            ca.Created_at
        FROM Citizen_Applications ca
        INNER JOIN Services s ON ca.ServiceId = s.ServiceId
        CROSS APPLY OPENJSON(ca.FormDetails, '$.Location')
            WITH (
                name VARCHAR(50) '$.name',
                value INT '$.value'
            ) AS loc
        LEFT JOIN TSWOTehsil t ON loc.name = 'Tehsil' AND t.TehsilId = loc.value
        LEFT JOIN District d ON loc.name = 'District' AND d.DistrictId = loc.value
        WHERE (
            (loc.name = 'Tehsil' AND t.DivisionCode = @AccessCode)
            OR (loc.name = 'District' AND d.Division = @AccessCode)
        )
          AND (@DataType IS NULL OR ca.DataType = @DataType)
          AND (@Department IS NULL OR s.DepartmentId = @Department);
    END
    ELSE IF @AccessLevel = 'District'
    BEGIN
        SELECT
            ca.ReferenceNumber,
            ca.FormDetails,
            ca.Created_at
        FROM Citizen_Applications ca
        INNER JOIN Services s ON ca.ServiceId = s.ServiceId
        CROSS APPLY OPENJSON(ca.FormDetails, '$.Location')
            WITH (
                name VARCHAR(50) '$.name',
                value INT '$.value'
            ) AS loc
        LEFT JOIN TSWOTehsil t ON loc.name = 'Tehsil' AND t.TehsilId = loc.value
        WHERE (
            (loc.name = 'Tehsil' AND t.DistrictId = @AccessCode)
            OR (loc.name = 'District' AND loc.value = @AccessCode)
        )
          AND (@DataType IS NULL OR ca.DataType = @DataType)
          AND (@Department IS NULL OR s.DepartmentId = @Department);
    END
END;

BEGIN

   -- Step 1: Calculate TotalRecords (apply all filters except pagination)
SELECT @TotalRecords = COUNT(DISTINCT ca.ReferenceNumber)

CROSS APPLY OPENJSON(ca.WorkFlow) AS wf
OUTER APPLY OPENJSON(ca.FormDetails, '$.Location')

                OR (@AccessLevel = 'Division' AND jsonLocation.name = 'District' AND d.Division = @AccessCode)
            )
           AND (
@ApplicationStatus = 'total'
            OR
            (
                JSON_VALUE(wf.value, '$.status') = @ApplicationStatus
                AND
                (@ApplicationStatus != 'sanctioned' OR ca.Status = 'Sanctioned')
            )
            OR
            (
                @ApplicationStatus = 'pensionstopped'
                AND ca.DataType = 'legacy'

            AND (@DataType IS NULL OR ca.DataType = @DataType)

CREATE PROCEDURE [dbo].[GetApplicationsForReport]
    @AccessCode INT,
    @ServiceId INT,
    @AccessLevel NVARCHAR(10), -- 'District', 'Tehsil', or 'All',
    @DataType NVARCHAR(10) = NULL
AS
BEGIN
    SET NOCOUNT ON;

    IF @AccessLevel = 'All'
    BEGIN
        SELECT
            t.TehsilName,
            COUNT(DISTINCT apps.ReferenceNumber) AS TotalApplicationsSubmitted,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'pending' THEN 1 ELSE 0 END), 0) AS TotalApplicationsPending,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'returntoedit' THEN 1 ELSE 0 END), 0) AS TotalApplicationsReturnToEdit,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'rejected' THEN 1 ELSE 0 END), 0) AS TotalApplicationsRejected,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'sanctioned' THEN 1 ELSE 0 END), 0) AS TotalApplicationsSanctioned
        FROM [dbo].[TSWOTehsil] t
        LEFT JOIN (
            SELECT
                ca.ReferenceNumber,
                ca.WorkFlow,
                jsonTehsil.value AS TehsilId
            FROM [dbo].[Citizen_Applications] ca
            CROSS APPLY OPENJSON(ca.FormDetails, '$.Location')
                WITH (name NVARCHAR(50) '$.name', value INT '$.value') AS jsonTehsil
            WHERE ca.ServiceId = @ServiceId
              AND jsonTehsil.name = 'Tehsil'
              AND (@DataType IS NULL OR ca.DataType = @DataType) -- ðŸ”¹ Added filter
        ) AS apps ON apps.TehsilId = t.TehsilId
        OUTER APPLY OPENJSON(apps.WorkFlow) AS wf
        GROUP BY t.TehsilName
        ORDER BY t.TehsilName;
    END
    ELSE IF @AccessLevel = 'District'
    BEGIN
        SELECT
            t.TehsilName,
            COUNT(DISTINCT apps.ReferenceNumber) AS TotalApplicationsSubmitted,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'pending' THEN 1 ELSE 0 END), 0) AS TotalApplicationsPending,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'returntoedit' THEN 1 ELSE 0 END), 0) AS TotalApplicationsReturnToEdit,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'rejected' THEN 1 ELSE 0 END), 0) AS TotalApplicationsRejected,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'sanctioned' THEN 1 ELSE 0 END), 0) AS TotalApplicationsSanctioned
        FROM [dbo].[TSWOTehsil] t
        LEFT JOIN (
            SELECT
                ca.ReferenceNumber,
                ca.WorkFlow,
                jsonTehsil.value AS TehsilId
            FROM [dbo].[Citizen_Applications] ca
            CROSS APPLY OPENJSON(ca.FormDetails, '$.Location')
                WITH (name NVARCHAR(50) '$.name', value INT '$.value') AS jsonTehsil
            WHERE ca.ServiceId = @ServiceId
              AND jsonTehsil.name = 'Tehsil'
              AND (@DataType IS NULL OR ca.DataType = @DataType) -- ðŸ”¹ Added filter
        ) AS apps ON apps.TehsilId = t.TehsilId
        OUTER APPLY OPENJSON(apps.WorkFlow) AS wf
        WHERE t.DistrictId = @AccessCode
        GROUP BY t.TehsilName
        ORDER BY t.TehsilName;
    END
    ELSE IF @AccessLevel = 'Tehsil'
    BEGIN
        SELECT
            COALESCE(t.TehsilName, 'Unknown') AS TehsilName,
            COUNT(DISTINCT apps.ReferenceNumber) AS TotalApplicationsSubmitted,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'pending' THEN 1 ELSE 0 END), 0) AS TotalApplicationsPending,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'returntoedit' THEN 1 ELSE 0 END), 0) AS TotalApplicationsReturnToEdit,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'rejected' THEN 1 ELSE 0 END), 0) AS TotalApplicationsRejected,
            ISNULL(SUM(CASE WHEN JSON_VALUE(wf.value, '$.status') = 'sanctioned' THEN 1 ELSE 0 END), 0) AS TotalApplicationsSanctioned
        FROM [dbo].[TSWOTehsil] t
        LEFT JOIN (
            SELECT
        ca.ReferenceNumber,
                ca.WorkFlow,
                jsonTehsil.value AS TehsilId
            FROM [dbo].[Citizen_Applications] ca
            CROSS APPLY OPENJSON(ca.FormDetails, '$.Location')
                WITH (name NVARCHAR(50) '$.name', value INT '$.value') AS jsonTehsil
            WHERE ca.ServiceId = @ServiceId
              AND jsonTehsil.name = 'Tehsil'
              AND jsonTehsil.value = @AccessCode
              AND (@DataType IS NULL OR ca.DataType = @DataType) -- ðŸ”¹ Added filter
        ) AS apps ON apps.TehsilId = t.TehsilId
        OUTER APPLY OPENJSON(apps.WorkFlow) AS wf
        WHERE t.TehsilId = @AccessCode
        GROUP BY t.TehsilName;
    END
END;

CREATE PROCEDURE [dbo].[GetCategoryCountsFromJson]
    @JsonKey NVARCHAR(100),
    @JsonPath NVARCHAR(200),
    @Categories CategoryList READONLY,
    @AccessLevel VARCHAR(20) = NULL,
    @AccessCode INT = NULL,
    @DivisionCode INT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @sql NVARCHAR(MAX) = N'
        SELECT
            c.Category AS ' + QUOTENAME(@JsonKey) + ',
            COUNT(DISTINCT a.ReferenceNumber) AS Count
        FROM @Categories c
        LEFT JOIN (
            SELECT DISTINCT ca.ReferenceNumber, pt.value AS Val
            FROM Citizen_Applications ca
            CROSS APPLY OPENJSON(ca.FormDetails, @JsonPath)
                WITH (value NVARCHAR(100) ''$.value'') pt
            WHERE ca.Status = ''Sanctioned''
              AND ca.DataType = ''new''
              AND (@AccessLevel = ''State'' OR 1 =
                  CASE
                      WHEN @AccessLevel = ''State'' THEN 1
                      WHEN EXISTS (
                          SELECT 1
                          FROM OPENJSON(ca.FormDetails, ''$.Location'')
                          WITH (name NVARCHAR(50) ''$.name'', value INT ''$.value'') loc
                          LEFT JOIN District d ON loc.name = ''District'' AND loc.value = d.DistrictID
                          LEFT JOIN Tehsil t ON loc.name = ''Tehsil'' AND loc.value = t.TehsilID
                          LEFT JOIN District d2 ON t.DistrictID = d2.DistrictID
                          WHERE
                              (@AccessLevel = ''District'' AND loc.name = ''District'' AND loc.value = @AccessCode)
                              OR (@AccessLevel = ''Tehsil'' AND loc.name = ''Tehsil'' AND loc.value = @AccessCode)
                              OR (@AccessLevel = ''Division'' AND (
                                  (loc.name = ''District'' AND d.Division = @DivisionCode)
                                  OR (loc.name = ''Tehsil'' AND d2.Division = @DivisionCode)
                              ))
                      ) THEN 1
                      ELSE 0
                  END
              )
        ) a ON c.Category = a.Val
        GROUP BY c.Category
        ORDER BY c.Category;
    ';

    EXEC sp_executesql
        @sql,
        N'@Categories CategoryList READONLY, @AccessLevel VARCHAR(20), @AccessCode INT, @DivisionCode INT',
        @Categories = @Categories,
        @AccessLevel = @AccessLevel,
        @AccessCode = @AccessCode,
        @DivisionCode = @DivisionCode;
END;
GO

        FROM OPENJSON(c.[Location])
        WITH (
            name NVARCHAR(50) '$.name',
            value INT '$.value'
        )
    ) AS jsonLocation
    LEFT JOIN [dbo].[District] d
        ON jsonLocation.name = 'District' AND jsonLocation.value = d.DistrictID
    OUTER APPLY (
        SELECT TOP 1
            JSON_VALUE(value, '$.status') AS OfficerStatus
        FROM OPENJSON(c.WorkFlow)
        WHERE JSON_VALUE(value, '$.designation') = @OfficerRole
    ) wf -- extract officer's own step status
    WHERE
        (
            @OfficerAccessLevel = 'State'
            OR (@OfficerAccessLevel = 'District' AND jsonLocation.name = 'District' AND jsonLocation.value = @OfficerAccessCode)
            OR (@OfficerAccessLevel = 'Tehsil' AND jsonLocation.name = 'Tehsil' AND jsonLocation.value = @OfficerAccessCode)
            OR (@OfficerAccessLevel = 'Division' AND jsonLocation.name = 'District' AND d.Division = @OfficerAccessCode)
        )
        AND (@ReferenceNumber IS NULL OR c.ReferenceNumber = @ReferenceNumber)
        AND (@CorrigendumId IS NULL OR c.CorrigendumId = @CorrigendumId)
        AND (
            @Status IS NULL
            OR @Status = 'total'
            OR wf.OfficerStatus = @Status
        )
        AND (@Type IS NULL OR c.Type = @Type);
END;

CREATE PROCEDURE [dbo].[GetCountForAdmin]
    @AccessLevel VARCHAR(50), -- 'State', 'Division', or 'District'
    @AccessCode INT, -- 0 for State, or actual DivisionId/DistrictId
    @DataType NVARCHAR(10) = NULL, -- optional filter
    @Department INT = NULL -- department filter (nullable)
AS
BEGIN
    SET NOCOUNT ON;

    -- Declare variables to store counts
    DECLARE @TotalOfficers INT = 0;
    DECLARE @TotalCitizens INT = 0;
    DECLARE @TotalApplicationsSubmitted INT = 0;
    DECLARE @TotalServices INT = 0;

    -- Validate AccessLevel
    IF @AccessLevel NOT IN ('State', 'Division', 'District')
    BEGIN
        SELECT
            -1 AS TotalOfficers,
            -1 AS TotalCitizens,
            -1 AS TotalApplicationsSubmitted,
            -1 AS TotalServices;
        RETURN;
    END;

    -- âœ… TotalServices (filter only if Department provided)
    SELECT @TotalServices = COUNT(*)
    FROM Services
    WHERE (@Department IS NULL OR DepartmentId = @Department);

    IF @AccessLevel = 'State'
    BEGIN
        -- Officers
        SELECT @TotalOfficers = COUNT(*)
        FROM Users u
        CROSS APPLY OPENJSON(u.AdditionalDetails)
            WITH (Department INT '$.Department') AS ad
        WHERE u.UserType = 'Officer'
          AND (@Department IS NULL OR ad.Department = @Department);

        -- Citizens
        SELECT @TotalCitizens = COUNT(*)
        FROM Users
        WHERE UserType = 'Citizen';

        -- Applications
        SELECT @TotalApplicationsSubmitted = COUNT(*)
        FROM Citizen_Applications ca
        INNER JOIN Services s ON ca.ServiceId = s.ServiceId
        WHERE (@Department IS NULL OR s.DepartmentId = @Department)
          AND (@DataType IS NULL OR ca.DataType = @DataType);
    END
    ELSE IF @AccessLevel = 'Division'
    BEGIN
        -- Officers
        SELECT @TotalOfficers = COUNT(DISTINCT u.UserId)
        FROM Users u
        CROSS APPLY OPENJSON(u.AdditionalDetails)
            WITH (
                AccessLevel VARCHAR(50) '$.AccessLevel',
                AccessCode INT '$.AccessCode',
                Department INT '$.Department'
            ) AS ad
        LEFT JOIN District d ON d.DistrictId = ad.AccessCode AND ad.AccessLevel = 'District'
        LEFT JOIN TSWOTehsil t ON t.TehsilId = ad.AccessCode AND ad.AccessLevel = 'Tehsil'
        WHERE u.UserType = 'Officer'
          AND (@Department IS NULL OR ad.Department = @Department)
          AND (
              (ad.AccessLevel = 'Division' AND ad.AccessCode = @AccessCode)
              OR (ad.AccessLevel = 'District' AND d.Division = @AccessCode)
              OR (ad.AccessLevel = 'Tehsil' AND t.DivisionCode = @AccessCode)
          );

        -- Citizens
        SELECT @TotalCitizens = COUNT(DISTINCT u.UserId)
        FROM Users u
        CROSS APPLY OPENJSON(u.AdditionalDetails)
            WITH (District INT '$.District', Tehsil INT '$.Tehsil') AS ad
        LEFT JOIN District d ON d.DistrictId = ad.District
        LEFT JOIN TSWOTehsil t ON t.TehsilId = ad.Tehsil
        WHERE u.UserType = 'Citizen'
          AND (d.Division = @AccessCode OR t.DivisionCode = @AccessCode);

        -- Applications
        SELECT @TotalApplicationsSubmitted = COUNT(DISTINCT ca.ReferenceNumber)
        FROM Citizen_Applications ca
        INNER JOIN Services s ON ca.ServiceId = s.ServiceId
        CROSS APPLY OPENJSON(ca.FormDetails, '$.Location')
            WITH (name VARCHAR(50) '$.name', value INT '$.value') AS loc
        LEFT JOIN TSWOTehsil t ON loc.name = 'Tehsil' AND t.TehsilId = loc.value
        LEFT JOIN District d ON loc.name = 'District' AND d.DistrictId = loc.value
        WHERE (@Department IS NULL OR s.DepartmentId = @Department)
          AND (
              (loc.name = 'Tehsil' AND t.DivisionCode = @AccessCode)
              OR (loc.name = 'District' AND d.Division = @AccessCode)
          )
  AND (@DataType IS NULL OR ca.DataType = @DataType);
    END
    ELSE IF @AccessLevel = 'District'
    BEGIN
        -- Officers
        SELECT @TotalOfficers = COUNT(DISTINCT u.UserId)
        FROM Users u
        CROSS APPLY OPENJSON(u.AdditionalDetails)
            WITH (
                AccessLevel VARCHAR(50) '$.AccessLevel',
                AccessCode INT '$.AccessCode',
                Department INT '$.Department'
            ) AS ad
        LEFT JOIN TSWOTehsil t ON t.TehsilId = ad.AccessCode AND ad.AccessLevel = 'Tehsil'
        WHERE u.UserType = 'Officer'
          AND (@Department IS NULL OR ad.Department = @Department)
          AND (
              (ad.AccessLevel = 'District' AND ad.AccessCode = @AccessCode)
              OR (ad.AccessLevel = 'Tehsil' AND t.DistrictId = @AccessCode)
          );

        -- Citizens
        SELECT @TotalCitizens = COUNT(DISTINCT u.UserId)
        FROM Users u
        CROSS APPLY OPENJSON(u.AdditionalDetails)
            WITH (District INT '$.District', Tehsil INT '$.Tehsil') AS ad
        LEFT JOIN TSWOTehsil t ON t.TehsilId = ad.Tehsil
        WHERE u.UserType = 'Citizen'
          AND (ad.District = @AccessCode OR t.DistrictId = @AccessCode);

        -- Applications
        SELECT @TotalApplicationsSubmitted = COUNT(DISTINCT ca.ReferenceNumber)
        FROM Citizen_Applications ca
        INNER JOIN Services s ON ca.ServiceId = s.ServiceId
        CROSS APPLY OPENJSON(ca.FormDetails, '$.Location')
            WITH (name VARCHAR(50) '$.name', value INT '$.value') AS loc
        LEFT JOIN TSWOTehsil t ON loc.name = 'Tehsil' AND t.TehsilId = loc.value
        WHERE (@Department IS NULL OR s.DepartmentId = @Department)
          AND (
              (loc.name = 'Tehsil' AND t.DistrictId = @AccessCode)
              OR (loc.name = 'District' AND loc.value = @AccessCode)
          )
          AND (@DataType IS NULL OR ca.DataType = @DataType);
    END

    -- Final Result
    SELECT
        @TotalOfficers AS TotalOfficers,
        @TotalCitizens AS TotalCitizens,
        @TotalApplicationsSubmitted AS TotalApplicationsSubmitted,
        @TotalServices AS TotalServices;
END;
GO

CREATE PROCEDURE [dbo].[GetDisabilityApplications]
    @AccessLevel VARCHAR(20),
    @AccessCode INT,
    @ServiceId INT,
    @TakenBy VARCHAR(50),
    @DivisionCode INT = NULL,
    @ResultType VARCHAR(20), -- 'totalpcpapplication' or 'expiringeligibility'
    @PageNumber INT = 1,
    @PageSize INT = 10
AS
BEGIN
    SET NOCOUNT ON;

    -- Validate pagination parameters
    IF @PageNumber < 1 SET @PageNumber = 1;
    IF @PageSize < 1 SET @PageSize = 10;

    -- ======== Ranked Workflows Section ========
    WITH RankedStatus AS (
        SELECT
            ca.*, -- Select all columns from Citizen_Applications
            jsonWorkFlow.status AS WorkflowStatus,
            ROW_NUMBER() OVER (
                PARTITION BY ca.ReferenceNumber
                ORDER BY COALESCE(jsonWorkFlow.timestamp, '9999-12-31') DESC, jsonWorkFlow.seq DESC
            ) AS rn
        FROM [dbo].[Citizen_Applications] ca
        CROSS APPLY OPENJSON(ca.WorkFlow) WITH (
            status NVARCHAR(50) '$.status',
            timestamp DATETIME '$.timestamp',
            seq INT '$.seq'
        ) AS jsonWorkFlow
        WHERE
            jsonWorkFlow.status IS NOT NULL
            AND jsonWorkFlow.status <> ''
            AND ca.ServiceId = @ServiceId
            AND ISJSON(ca.WorkFlow) = 1
    ),
    LatestCorrigendum AS (
        SELECT
            c.ReferenceNumber,
            c.CorrigendumFields,
            ROW_NUMBER() OVER (
                PARTITION BY c.ReferenceNumber
                ORDER BY c.CreatedAt DESC
            ) AS rn
        FROM [dbo].[Corrigendum] c
        WHERE
            ISJSON(c.CorrigendumFields) = 1
            AND EXISTS (
                SELECT 1
                FROM OPENJSON(c.CorrigendumFields) WITH (
                    new_value NVARCHAR(100) '$.IfTemporaryDisabilityUdidCardValidUpto.new_value'
                ) AS corr
                WHERE corr.new_value IS NOT NULL
            )
    ),
    LatestStatus AS (
        SELECT
            rs.*,
            lc.CorrigendumFields
        FROM RankedStatus rs
        LEFT JOIN LatestCorrigendum lc
            ON rs.ReferenceNumber = lc.ReferenceNumber
            AND lc.rn = 1
        WHERE rs.rn = 1
    ),
    FilteredApplications AS (
        SELECT
            ls.*, -- Carry forward all columns from LatestStatus
            CASE
                WHEN @ResultType = 'expiringeligibility' THEN
                    CASE
                        WHEN EXISTS (
                            SELECT 1
                            FROM OPENJSON(ls.FormDetails, '$."Pension Type"') WITH (
                                value NVARCHAR(100) '$.value',
                                additionalFields NVARCHAR(MAX) '$.additionalFields' AS JSON
                            ) AS pension
                            CROSS APPLY OPENJSON(pension.additionalFields) WITH (
                                name NVARCHAR(100) '$.name',
                                value NVARCHAR(100) '$.value',
                                additionalFields NVARCHAR(MAX) '$.additionalFields' AS JSON
                            ) AS disability
                            CROSS APPLY OPENJSON(disability.additionalFields) WITH (
                                name NVARCHAR(100) '$.name',
                                value NVARCHAR(100) '$.value'
                            ) AS tempDisability
                            OUTER APPLY (
                                SELECT new_value
                                FROM OPENJSON(ls.CorrigendumFields) WITH (
                                    new_value NVARCHAR(100) '$.IfTemporaryDisabilityUdidCardValidUpto.new_value'
                                )
                            ) AS corr
                            WHERE pension.value = 'PHYSICALLY CHALLENGED PERSON'
                                AND disability.name = 'KindOfDisability'
       AND disability.value = 'TEMPORARY'
                                AND tempDisability.name = 'IfTemporaryDisabilityUdidCardValidUpto'
                                AND (
                                    (corr.new_value IS NOT NULL
                                     AND TRY_CONVERT(DATE, corr.new_value) IS NOT NULL
                                     AND DATEDIFF(DAY, GETDATE(), TRY_CONVERT(DATE, corr.new_value)) BETWEEN 0 AND 90)
                                    OR
                                    (corr.new_value IS NULL
                                     AND tempDisability.value IS NOT NULL
                                     AND TRY_CONVERT(DATE, tempDisability.value) IS NOT NULL
                                     AND DATEDIFF(DAY, GETDATE(), TRY_CONVERT(DATE, tempDisability.value)) BETWEEN 0 AND 90)
                                )
                        ) THEN 1
                        ELSE 0
                    END
                ELSE 1 -- For totalpcpapplication, include all PHYSICALLY CHALLENGED PERSON applications
            END AS IsMatch
        FROM LatestStatus ls
        CROSS APPLY OPENJSON(ls.FormDetails, '$.Location') WITH (
            name NVARCHAR(50) '$.name',
            value INT '$.value'
        ) AS jsonLocation
        LEFT JOIN [dbo].[District] d
            ON jsonLocation.name = 'District'
            AND jsonLocation.value = d.DistrictID
        WHERE
            ls.ServiceId = @ServiceId
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District'
                    AND jsonLocation.name = 'District'
                    AND jsonLocation.value = @AccessCode)
                OR (@AccessLevel = 'Tehsil'
                    AND jsonLocation.name = 'Tehsil'
                    AND jsonLocation.value = @AccessCode)
                OR (
                    @AccessLevel = 'Division'
                    AND (
                        (jsonLocation.name = 'District'
                         AND jsonLocation.value = @AccessCode
                         AND d.Division = @DivisionCode)
                        OR
                        (jsonLocation.name = 'Tehsil'
                         AND EXISTS (
                            SELECT 1
                            FROM [dbo].[Tehsil] t
                            INNER JOIN [dbo].[District] d2
                                ON t.DistrictID = d2.DistrictID
                            WHERE t.TehsilID = jsonLocation.value
                                AND d2.Division = @DivisionCode
                         ))
                    )
                )
            )
            AND (
                @ResultType = 'totalpcpapplication' AND EXISTS (
                    SELECT 1
                    FROM OPENJSON(ls.FormDetails, '$."Pension Type"') WITH (
                        value NVARCHAR(100) '$.value'
                    ) AS pension
                    WHERE pension.value = 'PHYSICALLY CHALLENGED PERSON'
                )
                OR @ResultType = 'expiringeligibility'
            )
    )
    -- ======== Final SELECT with Pagination ========
    SELECT
        fa.*,
        CASE
            WHEN @ResultType = 'expiringeligibility' THEN
                COALESCE(
                    (SELECT corr.new_value
                     FROM OPENJSON(fa.CorrigendumFields) WITH (
                         new_value NVARCHAR(100) '$.IfTemporaryDisabilityUdidCardValidUpto.new_value'
                     ) AS corr
                     WHERE corr.new_value IS NOT NULL
                         AND TRY_CONVERT(DATE, corr.new_value) IS NOT NULL
                         AND DATEDIFF(DAY, GETDATE(), TRY_CONVERT(DATE, corr.new_value)) BETWEEN 0 AND 90),
                    (SELECT tempDisability.value
                     FROM OPENJSON(fa.FormDetails, '$."Pension Type"') WITH (
                         value NVARCHAR(100) '$.value',
                         additionalFields NVARCHAR(MAX) '$.additionalFields' AS JSON
                     ) AS pension
     CROSS APPLY OPENJSON(pension.additionalFields) WITH (
                         name NVARCHAR(100) '$.name',
                         value NVARCHAR(100) '$.value',
                         additionalFields NVARCHAR(MAX) '$.additionalFields' AS JSON
                     ) AS disability
                     CROSS APPLY OPENJSON(disability.additionalFields) WITH (
                         name NVARCHAR(100) '$.name',
                         value NVARCHAR(100) '$.value'
                     ) AS tempDisability
                     WHERE pension.value = 'PHYSICALLY CHALLENGED PERSON'
                         AND disability.name = 'KindOfDisability'
                         AND disability.value = 'TEMPORARY'
                         AND tempDisability.name = 'IfTemporaryDisabilityUdidCardValidUpto'
                         AND fa.IsMatch = 1)
                )
            ELSE NULL
        END AS Expiration_Date
    FROM FilteredApplications fa
    WHERE fa.IsMatch = 1
    ORDER BY fa.ReferenceNumber
    OFFSET (@PageNumber - 1) * @PageSize ROWS
    FETCH NEXT @PageSize ROWS ONLY;
END;
GO

CREATE PROCEDURE [dbo].[GetDuplicateAccNo]
    @AccountNumber VARCHAR(50),
    @BankName VARCHAR(100),
    @IfscCode VARCHAR(11)
AS
BEGIN
    SET NOCOUNT ON;

    -- This CTE extracts bank info regardless of the key name: "Bank Details" or "BankDetails"
    WITH MatchingAccounts AS (
        SELECT
            ca.*
        FROM
            [dbo].[Citizen_Applications] ca

        OUTER APPLY (
            SELECT
                MAX(CASE WHEN bd.name = 'AccountNumber' THEN bd.value END) AS AccountNumber,
                MAX(CASE WHEN bd.name = 'BankName' THEN bd.value END) AS BankName,
                MAX(CASE WHEN bd.name = 'IfscCode' THEN bd.value END) AS IfscCode
            FROM (
                -- Try parsing both keys
                SELECT * FROM OPENJSON(ca.FormDetails, '$."Bank Details"')
                         WITH (name NVARCHAR(50) '$.name', value NVARCHAR(100) '$.value')
                UNION ALL
                SELECT * FROM OPENJSON(ca.FormDetails, '$.BankDetails')
                         WITH (name NVARCHAR(50) '$.name', value NVARCHAR(100) '$.value')
            ) AS bd
        ) AS bankInfo

        WHERE
            bankInfo.AccountNumber = @AccountNumber
            AND bankInfo.IfscCode = @IfscCode
            AND (
                @BankName IS NULL
                OR bankInfo.BankName LIKE '%' + @BankName + '%'
            )
    )

    SELECT
        *
    FROM
        MatchingAccounts;
END;
GO

CREATE PROCEDURE [dbo].[GetGenderCounts]
    @ServiceId INT,
    @AccessLevel VARCHAR(20),
    @AccessCode INT,
    @ApplicationStatus VARCHAR(50) = NULL,
    @DataType VARCHAR(20) = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- Step 1: Extract Gender and filter
    WITH Gender_CTE AS
    (
        SELECT
            ca.ReferenceNumber,
            JSON_VALUE(ad.value, '$.value') AS Gender
        FROM dbo.Citizen_Applications ca

        -- Extract Gender from Applicant Details
        OUTER APPLY OPENJSON(ca.FormDetails, '$."Applicant Details"') AS ad

        -- Extract Location for AccessLevel filtering
        OUTER APPLY OPENJSON(ca.FormDetails, '$.Location')
            WITH (name NVARCHAR(50) '$.name', value INT '$.value') AS loc

        LEFT JOIN [dbo].[District] d
            ON loc.name = 'District' AND loc.value = d.DistrictID

        WHERE
            ca.ServiceId = @ServiceId
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District' AND loc.name = 'District' AND loc.value = @AccessCode)
                OR (@AccessLevel = 'Tehsil' AND loc.name = 'Tehsil' AND loc.value = @AccessCode)
                OR (@AccessLevel = 'Division' AND loc.name = 'District' AND d.Division = @AccessCode)
            )
            AND (
                @ApplicationStatus = 'total'
                OR (@ApplicationStatus IS NOT NULL AND ca.Status = @ApplicationStatus)
            )
            AND (@DataType IS NULL OR ca.DataType = @DataType)
            AND JSON_VALUE(ad.value, '$.name') = 'Gender'
            AND JSON_VALUE(ad.value, '$.value') IS NOT NULL
    )

    -- Step 2: Return count grouped by Gender
    SELECT
        Gender,
        COUNT(*) AS CountOfApplicants
    FROM Gender_CTE
    GROUP BY Gender
    ORDER BY Gender;
END;
GO

CREATE PROCEDURE [dbo].[GetInitiatedApplications]
    @UserId INT,
    @PageIndex INT = 0,
    @PageSize INT = 10,
    @IsPaginated BIT = 1,
    @TotalRecords INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    -- Calculate total records before pagination
    SET @TotalRecords = (
        SELECT COUNT(*)
        FROM Citizen_Applications ca
        WHERE ca.Citizen_id = @UserId
        AND ca.Status <> 'Incomplete'
    );

    -- CTE to assign row numbers for pagination
    WITH OrderedApplications AS
    (
        SELECT
            ca.ReferenceNumber,
            ca.ReferenceNumberAlphaNumeric,
            ca.Citizen_id,
            ca.ServiceId,
            ca.DistrictUidForBank,
            ca.FormDetails,
            ca.WorkFlow,
            ca.AdditionalDetails,
            ca.CurrentPlayer,
            ca.Status,
            ca.DataType,
            ca.Created_at,
            ROW_NUMBER() OVER (
                ORDER BY
                    TRY_CAST(PARSENAME(REPLACE(ca.ReferenceNumber, '/', '.'), 1) AS INT)
            ) AS RowNum
        FROM
            Citizen_Applications ca
        WHERE
            ca.Citizen_id = @UserId
            AND ca.Status <> 'Incomplete'
    )

    SELECT
        oa.ReferenceNumber,
        oa.ReferenceNumberAlphaNumeric,
        oa.Citizen_id,
        oa.ServiceId,
        oa.DistrictUidForBank,
        oa.FormDetails,
        oa.WorkFlow,
        oa.AdditionalDetails,
        oa.CurrentPlayer,
        oa.Status,
        oa.DataType,
        oa.Created_at
    FROM
        OrderedApplications oa
    WHERE
        (@IsPaginated = 1 AND oa.RowNum BETWEEN (@PageIndex * @PageSize) + 1 AND ((@PageIndex + 1) * @PageSize))
        OR (@IsPaginated = 0)
    ORDER BY
        oa.RowNum;
END;
GO

CREATE PROCEDURE [dbo].[GetLegacyStatusCount]
    @AccessLevel VARCHAR(20),
    @AccessCode INT,
    @ServiceId INT,
    @DivisionCode INT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- Step 1: Filter and deduplicate applications
    WITH RankedApplications AS (
        SELECT
            ca.ReferenceNumber,
            ca.Status,
            ROW_NUMBER() OVER (PARTITION BY ca.ReferenceNumber ORDER BY ca.Created_at DESC) AS rn
        FROM [dbo].[Citizen_Applications] ca
        CROSS APPLY OPENJSON(ca.FormDetails, '$.Location')
            WITH (
                name NVARCHAR(50) '$.name',
                value INT '$.value'
            ) AS jsonLocation
        LEFT JOIN [dbo].[District] d
            ON jsonLocation.name = 'District'
            AND jsonLocation.value = d.DistrictID
        WHERE
            ca.ServiceId = @ServiceId
            AND ca.DataType = 'legacy'
            AND (
                @AccessLevel = 'State'

                OR (
                    @AccessLevel = 'Division'
                    AND jsonLocation.name = 'District'
                    AND EXISTS (
                        SELECT 1
                        FROM [dbo].[District] d1
                        WHERE d1.DistrictID = jsonLocation.value
                            AND d1.Division = @AccessCode
                    )
                )

                OR (
                    @AccessLevel = 'District'
                    AND jsonLocation.name = 'District'
                    AND jsonLocation.value = @AccessCode
                )

                OR (
                    @AccessLevel = 'Tehsil'
                    AND (
                        (jsonLocation.name = 'Tehsil' AND jsonLocation.value = @AccessCode)
                        OR
                        (jsonLocation.name = 'District'
                            AND EXISTS (
                                SELECT 1
                                FROM [dbo].[TSWOTehsil] tt
                                WHERE tt.TehsilId = @AccessCode
                                  AND tt.DistrictId = jsonLocation.value
                            )
                        )
                    )
                )
            )
    ),
    FilteredApplications AS (
        SELECT ReferenceNumber, Status
        FROM RankedApplications
        WHERE rn = 1 -- Take only one row per ReferenceNumber
    )

    -- Step 2: Final Status Summary
    SELECT
        COUNT(*) AS TotalApplications,
        ISNULL(SUM(CASE WHEN LOWER(Status) = 'sanctioned' THEN 1 ELSE 0 END), 0) AS SanctionedCount,
        ISNULL(SUM(CASE WHEN LOWER(Status) = 'rejected' THEN 1 ELSE 0 END), 0) AS RejectCount,
        ISNULL(SUM(CASE
            WHEN LOWER(Status) NOT IN ('sanctioned', 'rejected') THEN 1
            ELSE 0
        END), 0) AS PensionStoppedCount
    FROM FilteredApplications;
END;
GO

CREATE PROCEDURE [dbo].[GetLocationWiseSanctionedCounts]
    @AccessLevel VARCHAR(20) = NULL,
    @AccessCode INT = NULL,
    @DivisionCode INT = NULL,
    @ServiceId INT
AS
BEGIN
    SET NOCOUNT ON;

    BEGIN TRY
        DECLARE @sql NVARCHAR(MAX);

        IF @AccessLevel = 'State'
        BEGIN
            SET @sql = '
            SELECT
                COALESCE(DivisionID, 0) AS LocationId,
                COALESCE(DivisionName, ''Unknown Division'') AS LocationName,
                COUNT(DISTINCT ReferenceNumber) AS Count
            FROM (
                SELECT DISTINCT
                    ca.ReferenceNumber,
                    -- Prioritize Tehsil division over District division when both exist
                    COALESCE(
                        MAX(CASE WHEN jsonLocation.name = ''Tehsil'' THEN d2.Division END),
                        MAX(CASE WHEN jsonLocation.name = ''District'' THEN d.Division END)
                    ) AS DivisionID,
                    COALESCE(
                        CASE MAX(CASE WHEN jsonLocation.name = ''Tehsil'' THEN d2.Division END)
                            WHEN 1 THEN ''Jammu''
                            WHEN 2 THEN ''Kashmir''
                            ELSE NULL END,
                        CASE MAX(CASE WHEN jsonLocation.name = ''District'' THEN d.Division END)
                            WHEN 1 THEN ''Jammu''
                            WHEN 2 THEN ''Kashmir''
                            ELSE ''Unknown'' END
                    ) AS DivisionName
                FROM [dbo].[Citizen_Applications] ca
                CROSS APPLY OPENJSON(ca.FormDetails, ''$.Location'') WITH (
                    name NVARCHAR(50) ''$.name'',
                    value INT ''$.value''
                ) AS jsonLocation
                LEFT JOIN [dbo].[District] d
                    ON jsonLocation.name = ''District''
                    AND jsonLocation.value = d.DistrictID
                LEFT JOIN [dbo].[TSWOTehsil] t
                    ON jsonLocation.name = ''Tehsil''
                    AND jsonLocation.value = t.TehsilID
                LEFT JOIN [dbo].[District] d2
                    ON t.DistrictID = d2.DistrictID
                WHERE ca.ServiceId = @ServiceId
                    AND ca.Status = ''Sanctioned''
                    AND ca.DataType = ''new''
                    AND jsonLocation.name IN (''District'', ''Tehsil'')
                GROUP BY ca.ReferenceNumber
            ) sub
            WHERE DivisionID IS NOT NULL
            GROUP BY DivisionID, DivisionName
            ORDER BY DivisionName;
            ';
        END
        ELSE IF @AccessLevel = 'Division'
        BEGIN
            SET @sql = '
            SELECT
                COALESCE(DistrictID, 0) AS LocationId,
                COALESCE(DistrictName, ''Unknown District'') AS LocationName,
                COUNT(DISTINCT ReferenceNumber) AS Count
            FROM (
                SELECT DISTINCT
                    ca.ReferenceNumber,
                    -- Prioritize Tehsil district over District when both exist
                    COALESCE(
                        MAX(CASE WHEN jsonLocation.name = ''Tehsil'' THEN d2.DistrictID END),
                        MAX(CASE WHEN jsonLocation.name = ''District'' THEN d.DistrictID END)
                    ) AS DistrictID,
                    COALESCE(
                        MAX(CASE WHEN jsonLocation.name = ''Tehsil'' THEN d2.DistrictName END),
                        MAX(CASE WHEN jsonLocation.name = ''District'' THEN d.DistrictName END)
                    ) AS DistrictName,
                    -- Get division for filtering
                    COALESCE(
                        MAX(CASE WHEN jsonLocation.name = ''Tehsil'' THEN d2.Division END),
                        MAX(CASE WHEN jsonLocation.name = ''District'' THEN d.Division END)
                    ) AS DivisionID
                FROM [dbo].[Citizen_Applications] ca
                CROSS APPLY OPENJSON(ca.FormDetails, ''$.Location'') WITH (
                    name NVARCHAR(50) ''$.name'',
                    value INT ''$.value''
                ) AS jsonLocation
                LEFT JOIN [dbo].[District] d
                    ON jsonLocation.name = ''District''
                    AND jsonLocation.value = d.DistrictID
                LEFT JOIN [dbo].[TSWOTehsil] t
                    ON jsonLocation.name = ''Tehsil''
                    AND jsonLocation.value = t.TehsilID
                LEFT JOIN [dbo].[District] d2
                    ON t.DistrictID = d2.DistrictID
                WHERE ca.ServiceId = @ServiceId
                    AND ca.Status = ''Sanctioned''
                    AND ca.DataType = ''new''
                    AND jsonLocation.name IN (''District'', ''Tehsil'')
                GROUP BY ca.ReferenceNumber
                HAVING COALESCE(
                    MAX(CASE WHEN jsonLocation.name = ''Tehsil'' THEN d2.Division END),
                    MAX(CASE WHEN jsonLocation.name = ''District'' THEN d.Division END)
                ) = @DivisionCode
            ) sub
            WHERE DistrictID IS NOT NULL
            GROUP BY DistrictID, DistrictName
            ORDER BY DistrictName;
            ';
        END
        ELSE IF @AccessLevel = 'District'
        BEGIN
            SET @sql = '
            SELECT
                COALESCE(t.TehsilID, 0) AS LocationId,
                COALESCE(t.TehsilName, ''Unknown Tehsil'') AS LocationName,
                COUNT(DISTINCT ca.ReferenceNumber) AS Count
            FROM [dbo].[Citizen_Applications] ca
            CROSS APPLY OPENJSON(ca.FormDetails, ''$.Location'') WITH (
                name NVARCHAR(50) ''$.name'',
                value INT ''$.value''
            ) AS jsonLocation
            LEFT JOIN [dbo].[TSWOTehsil] t
                ON jsonLocation.name = ''Tehsil''
                AND jsonLocation.value = t.TehsilID
            WHERE ca.ServiceId = @ServiceId
                AND ca.Status = ''Sanctioned''
                AND ca.DataType = ''new''
                AND jsonLocation.name = ''Tehsil''
                AND t.DistrictID = @AccessCode
            GROUP BY t.TehsilID, t.TehsilName
            HAVING t.TehsilID IS NOT NULL
            ORDER BY t.TehsilName;
            ';
        END
        ELSE IF @AccessLevel = 'Tehsil'
        BEGIN
            SET @sql = '
            SELECT
                COALESCE(@AccessCode, 0) AS LocationId,
                COALESCE(
                    (SELECT TehsilName FROM [dbo].[TSWOTehsil] WHERE TehsilID = @AccessCode),
                    ''Unknown Tehsil''
                ) AS LocationName,
                COUNT(DISTINCT ca.ReferenceNumber) AS Count
            FROM [dbo].[Citizen_Applications] ca
            CROSS APPLY OPENJSON(ca.FormDetails, ''$.Location'') WITH (
                name NVARCHAR(50) ''$.name'',
                value INT ''$.value''
            ) AS jsonLocation
            WHERE ca.ServiceId = @ServiceId
                AND ca.Status = ''Sanctioned''
                AND ca.DataType = ''new''
                AND jsonLocation.name = ''Tehsil''
                AND jsonLocation.value = @AccessCode;
            ';
        END

        EXEC sp_executesql
            @sql,
            N'@AccessLevel VARCHAR(20), @AccessCode INT, @DivisionCode INT, @ServiceId INT',
            @AccessLevel = @AccessLevel,
            @AccessCode = @AccessCode,
            @DivisionCode = @DivisionCode,
            @ServiceId = @ServiceId;
    END TRY
    BEGIN CATCH
        DECLARE @ErrorMessage NVARCHAR(4000) = ERROR_MESSAGE();
        DECLARE @ErrorSeverity INT = ERROR_SEVERITY();
        DECLARE @ErrorState INT = ERROR_STATE();

        RAISERROR (@ErrorMessage, @ErrorSeverity, @ErrorState);
    END CATCH;
END;
GO

CREATE PROCEDURE [dbo].[GetMainApplicationStatusCount]
    @AccessLevel VARCHAR(20) = NULL,
    @AccessCode INT = NULL,
    @ServiceId INT,
    @TakenBy VARCHAR(50) = NULL,
    @DivisionCode INT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- ======== Main Applications Section ========
    WITH RankedStatus AS (
        SELECT
            ca.ReferenceNumber,
            ca.ServiceId,
            ca.FormDetails,
            LOWER(LTRIM(RTRIM(COALESCE(jsonWorkFlow.status, '')))) AS status,
            jsonWorkFlow.completedAtRaw,
            jsonWorkFlow.seq,
            jsonWorkFlow.designation,
            jsonWorkFlow.playerId,
            ca.CurrentPlayer,
            ROW_NUMBER() OVER (
                PARTITION BY ca.ReferenceNumber
                ORDER BY
                    CASE
                        WHEN @TakenBy IS NULL AND ca.CurrentPlayer = jsonWorkFlow.playerId THEN 1
                        WHEN @TakenBy IS NOT NULL AND jsonWorkFlow.designation = @TakenBy THEN 1
                        WHEN jsonWorkFlow.status = 'returntoedit' THEN 2
                        ELSE 3
                    END ASC, -- Prioritize CurrentPlayer=playerId when TakenBy is NULL
                    COALESCE(
                        TRY_CONVERT(DATETIME, jsonWorkFlow.completedAtRaw),
                        TRY_CONVERT(DATETIME, REPLACE(jsonWorkFlow.completedAtRaw, 'Z', '')), -- Handle ISO formats
                        TRY_CONVERT(DATETIME, REPLACE(jsonWorkFlow.completedAtRaw, 'T', ' ')), -- Handle T separator
                        CAST('9999-12-31T23:59:59.997' AS DATETIME)
                    ) DESC,
                    COALESCE(jsonWorkFlow.seq, 0) DESC
            ) AS rn
        FROM [dbo].[Citizen_Applications] ca
        CROSS APPLY OPENJSON(ca.WorkFlow) WITH (
            status NVARCHAR(50) '$.status',
            completedAtRaw NVARCHAR(50) '$.completedAt',
            seq INT '$.seq',
            designation NVARCHAR(50) '$.designation',
            playerId NVARCHAR(50) '$.playerId'
        ) AS jsonWorkFlow
        WHERE
            jsonWorkFlow.status IS NOT NULL
            AND jsonWorkFlow.status <> ''
            AND ca.ServiceId = @ServiceId
            AND ca.DataType = 'legacy'
            AND ISJSON(ca.WorkFlow) = 1
    ),
    LatestStatus AS (
        SELECT
            ReferenceNumber,
            ServiceId,
            FormDetails,
            status
        FROM RankedStatus
        WHERE rn = 1
    ),
    FilteredApplications AS (
        SELECT DISTINCT
            ls.ReferenceNumber,
            ls.ServiceId,
            ls.status,
            CASE
                WHEN EXISTS (
                    SELECT 1
                    FROM [dbo].[Corrigendum] c
                    WHERE c.ReferenceNumber = ls.ReferenceNumber
                        AND c.Type = 'Corrigendum'
                ) THEN 1
                ELSE 0
            END AS HasCorrigendum
        FROM LatestStatus ls
        OUTER APPLY OPENJSON(ls.FormDetails, '$.Location') WITH (
            name NVARCHAR(50) '$.name',
            value INT '$.value'
        ) AS jsonLocation
        LEFT JOIN [dbo].[District] d
            ON jsonLocation.name = 'District'
            AND jsonLocation.value = d.DistrictID
        WHERE ls.ServiceId = @ServiceId
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District'
                    AND jsonLocation.name = 'District'
                    AND jsonLocation.value = @AccessCode)
                OR (@AccessLevel = 'Tehsil'
                    AND jsonLocation.name = 'Tehsil'
                    AND jsonLocation.value = @AccessCode)
                OR (
                    @AccessLevel = 'Division'
                    AND (
                        (jsonLocation.name = 'District'
                            AND d.Division = @DivisionCode)
       OR
                        (jsonLocation.name = 'Tehsil'
                            AND EXISTS (
                                SELECT 1
                                FROM Tehsil t
                                INNER JOIN District d2
                                    ON t.DistrictID = d2.DistrictID
                                WHERE t.TehsilID = jsonLocation.value
                                    AND d2.Division = @DivisionCode
                            ))
                    )
                )
                OR jsonLocation.name IS NULL
            )
    )

    -- ======== Debugging Output (Optional) ========
    /*
    SELECT
        fa.ReferenceNumber,
        fa.status
    FROM FilteredApplications fa
    ORDER BY fa.ReferenceNumber;

    SELECT
        rs.ReferenceNumber,
        rs.status,
        rs.completedAtRaw,
        rs.seq,
        rs.designation,
        rs.CurrentPlayer,
        rs.playerId,
        CASE
            WHEN rs.CurrentPlayer = rs.playerId THEN 'Match'
            ELSE 'No Match'
        END AS PlayerMatch,
        rs.rn
    FROM RankedStatus rs
    WHERE rs.ServiceId = @ServiceId
    ORDER BY rs.ReferenceNumber, rs.rn;
    */

    -- ======== Final SELECT for Main Application Status Counts ========
    SELECT
        COUNT(DISTINCT CASE
            WHEN fa.status IN ('pending', 'forwarded', 'returned')
            THEN fa.ReferenceNumber END
        ) AS PendingCount,
        COUNT(DISTINCT CASE
            WHEN fa.status = 'returntoedit'
            THEN fa.ReferenceNumber END
        ) AS ReturnToEditCount,
        COUNT(DISTINCT CASE
            WHEN fa.status = 'sanctioned'
            THEN fa.ReferenceNumber END
        ) AS SanctionedCount,
        COUNT(DISTINCT CASE
            WHEN fa.status = 'rejected'
            THEN fa.ReferenceNumber END
        ) AS RejectCount,
        COUNT(DISTINCT fa.ReferenceNumber) AS TotalApplications
    FROM
        FilteredApplications fa;
END;
GO

CREATE PROCEDURE [dbo].[GetNextApplicationCount]
    @DistrictId INT,
    @ServiceId INT,
    @FinancialYear VARCHAR(50),
    @Type NVARCHAR(30) = NULL,
    @NextCount INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @OutputTable TABLE (CountValue INT);
    DECLARE @RetryCount INT = 0;
    DECLARE @MaxRetries INT = 3;

    WHILE @RetryCount < @MaxRetries
    BEGIN
        BEGIN TRY
            -- Use SERIALIZABLE to prevent concurrent reads/writes
            SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;
            BEGIN TRANSACTION;

            -- Try to update existing row
            UPDATE ApplicationPerDistrict
            SET CountValue = CountValue + 1
            OUTPUT inserted.CountValue INTO @OutputTable (CountValue)
            WHERE DistrictId = @DistrictId
              AND ServiceId = @ServiceId
              AND FinancialYear = @FinancialYear
              AND (@Type IS NULL OR Type = @Type);

            -- If no row was updated, insert a new one
            IF @@ROWCOUNT = 0
            BEGIN
                INSERT INTO ApplicationPerDistrict (Type, DistrictId, ServiceId, FinancialYear, CountValue)
                VALUES (@Type, @DistrictId, @ServiceId, @FinancialYear, 1);

                SET @NextCount = 1;
            END
            ELSE
            BEGIN
                -- Get the output value
                SELECT @NextCount = CountValue FROM @OutputTable;
            END

            COMMIT TRANSACTION;
            SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
            RETURN;
        END TRY
        BEGIN CATCH
            IF @@TRANCOUNT > 0
                ROLLBACK TRANSACTION;

            -- Handle deadlock or unique constraint violation
            IF ERROR_NUMBER() IN (1205, 2627, 2601) -- Deadlock or unique constraint
            BEGIN
                SET @RetryCount = @RetryCount + 1;
                IF @RetryCount < @MaxRetries
                BEGIN
                    WAITFOR DELAY '00:00:00.100'; -- Wait 100ms before retry
                    CONTINUE;
                END
            END

            -- Rethrow other errors
            DECLARE @ErrorMessage NVARCHAR(4000) = ERROR_MESSAGE();
            DECLARE @ErrorSeverity INT = ERROR_SEVERITY();
            DECLARE @ErrorState INT = ERROR_STATE();
            RAISERROR (@ErrorMessage, @ErrorSeverity, @ErrorState);
            RETURN;
        END CATCH
    END

    -- Max retries reached
    RAISERROR ('Failed to get next count after %d retries due to concurrency issues.', 16, 1, @MaxRetries);
END;
GO

CREATE PROCEDURE [dbo].[GetOfficersByAccessLevel]
    @AccessLevel VARCHAR(100),
    @AccessCode INT,
    @Department INT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    IF @AccessLevel = 'State'
    BEGIN
        SELECT
            Name,
            Username,
            Email,
            MobileNumber,
            JSON_VALUE(AdditionalDetails, '$.Role') AS Designation,
            JSON_VALUE(AdditionalDetails, '$.AccessLevel') AS AccessLevel,
            JSON_VALUE(AdditionalDetails, '$.AccessCode') AS AccessCode,
            JSON_VALUE(AdditionalDetails, '$.Validate') AS IsValidated
        FROM Users u
        CROSS APPLY OPENJSON(u.AdditionalDetails)
            WITH (
                Department INT '$.Department'
            ) AS ad
        WHERE UserType = 'Officer'
          AND (@Department IS NULL OR ad.Department = @Department);
    END
    ELSE IF @AccessLevel = 'Division'
    BEGIN
        SELECT
            u.Name,
            u.Username,
            u.Email,
            u.MobileNumber,
            JSON_VALUE(u.AdditionalDetails, '$.Role') AS Designation,
            JSON_VALUE(u.AdditionalDetails, '$.AccessLevel') AS AccessLevel,
            JSON_VALUE(u.AdditionalDetails, '$.AccessCode') AS AccessCode,
            JSON_VALUE(u.AdditionalDetails, '$.Validate') AS IsValidated
        FROM Users u
        CROSS APPLY OPENJSON(u.AdditionalDetails)
            WITH (
                AccessLevel VARCHAR(50) '$.AccessLevel',
                AccessCode INT '$.AccessCode',
                Department INT '$.Department'
            ) AS ad
        LEFT JOIN District d ON d.DistrictId = ad.AccessCode AND ad.AccessLevel = 'District'
        LEFT JOIN TSWOTehsil t ON t.TehsilId = ad.AccessCode AND ad.AccessLevel = 'Tehsil'
        WHERE u.UserType = 'Officer'
          AND (@Department IS NULL OR ad.Department = @Department)
          AND (
              (ad.AccessLevel = 'Division' AND ad.AccessCode = @AccessCode)
              OR (ad.AccessLevel = 'District' AND d.Division = @AccessCode)
              OR (ad.AccessLevel = 'Tehsil' AND t.DivisionCode = @AccessCode)
          );
    END
    ELSE IF @AccessLevel = 'District'
    BEGIN
        SELECT
            u.Name,
            u.Username,
            u.Email,
            u.MobileNumber,
            JSON_VALUE(u.AdditionalDetails, '$.Role') AS Designation,
            JSON_VALUE(u.AdditionalDetails, '$.AccessLevel') AS AccessLevel,
            JSON_VALUE(u.AdditionalDetails, '$.AccessCode') AS AccessCode,
            JSON_VALUE(u.AdditionalDetails, '$.Validate') AS IsValidated
        FROM Users u
        CROSS APPLY OPENJSON(u.AdditionalDetails)
            WITH (
                AccessLevel VARCHAR(50) '$.AccessLevel',
                AccessCode INT '$.AccessCode',
                Department INT '$.Department'
            ) AS ad
        LEFT JOIN TSWOTehsil t ON t.TehsilId = ad.AccessCode AND ad.AccessLevel = 'Tehsil'
        WHERE u.UserType = 'Officer'
          AND (@Department IS NULL OR ad.Department = @Department)
          AND (
              (ad.AccessLevel = 'District' AND ad.AccessCode = @AccessCode)
              OR (ad.AccessLevel = 'Tehsil' AND t.DistrictId = @AccessCode)
          );
    END
END;

CREATE PROCEDURE [dbo].[GetOfficersToValidate]
    @AccessLevel VARCHAR(100),
    @AccessCode INT
AS
BEGIN
    SET NOCOUNT ON;

    IF @AccessLevel = 'State'
    BEGIN
        SELECT
            [Name],
            Username,
            [Email],
            [MobileNumber],
            JSON_VALUE(AdditionalDetails, '$.Role') AS Designation,
            [UserType],
            TRY_CAST(JSON_VALUE(AdditionalDetails, '$.Validate') AS BIT) AS IsValidated
        FROM Users
        WHERE
            UserType = 'Officer';
    END

    IF @AccessLevel = 'Division'
    BEGIN
        -- Officers at Division level
        SELECT
            [Name],
            Username,
            [Email],
            [MobileNumber],
            JSON_VALUE(AdditionalDetails, '$.Role') AS Designation,
            [UserType],
            TRY_CAST(JSON_VALUE(AdditionalDetails, '$.Validate') AS BIT) AS IsValidated
        FROM Users
        WHERE
            UserType = 'Officer'
            AND JSON_VALUE(AdditionalDetails, '$.AccessLevel') = 'Division'
            AND TRY_CAST(JSON_VALUE(AdditionalDetails, '$.AccessCode') AS INT) = @AccessCode

        UNION ALL

        -- Officers at District level under this Division
        SELECT
            u.[Name],
            u.Username,
            u.[Email],
            u.[MobileNumber],
            JSON_VALUE(u.AdditionalDetails, '$.Role') AS Designation,
            u.[UserType],
            TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.Validate') AS BIT) AS IsValidated
        FROM Users u
        INNER JOIN District d ON
            TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.AccessLevel') AS VARCHAR) = 'District'
            AND TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.AccessCode') AS INT) = d.DistrictID
        WHERE
            u.UserType = 'Officer'
            AND d.Division = @AccessCode

        UNION ALL

        -- Officers at Tehsil level under this Division
        SELECT
            u.[Name],
            u.Username,
            u.[Email],
            u.[MobileNumber],
            JSON_VALUE(u.AdditionalDetails, '$.Role') AS Designation,
            u.[UserType],
            TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.Validate') AS BIT) AS IsValidated
        FROM Users u
        INNER JOIN TSWOTehsil t ON
            TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.AccessLevel') AS VARCHAR) = 'Tehsil'
            AND TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.AccessCode') AS INT) = t.TehsilId
        WHERE
            u.UserType = 'Officer'
            AND t.divisionCode = @AccessCode;
    END

    IF @AccessLevel = 'District'
    BEGIN
        -- Officers at District level
        SELECT
            [Name],
            Username,
            [Email],
            [MobileNumber],
            JSON_VALUE(AdditionalDetails, '$.Role') AS Designation,
            [UserType],
            TRY_CAST(JSON_VALUE(AdditionalDetails, '$.Validate') AS BIT) AS IsValidated
        FROM Users
        WHERE
            UserType = 'Officer'
            AND JSON_VALUE(AdditionalDetails, '$.AccessLevel') = 'District'
            AND TRY_CAST(JSON_VALUE(AdditionalDetails, '$.AccessCode') AS INT) = @AccessCode

        UNION ALL

        -- Officers at Tehsil level under this District
        SELECT
            u.[Name],
            u.Username,
            u.[Email],
            u.[MobileNumber],
            JSON_VALUE(u.AdditionalDetails, '$.Role') AS Designation,
            u.[UserType],
            TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.Validate') AS BIT) AS IsValidated
        FROM Users u
        INNER JOIN TSWOTehsil t ON
            TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.AccessLevel') AS VARCHAR) = 'Tehsil'
            AND TRY_CAST(JSON_VALUE(u.AdditionalDetails, '$.AccessCode') AS INT) = t.TehsilId
        WHERE
            u.UserType = 'Officer'
            AND t.DistrictID = @AccessCode;
 END

CREATE PROCEDURE [dbo].[GetRecordsForBankFile]
    @AccessCode INT,
    @ApplicationStatus VARCHAR(50) = NULL,
    @ServiceId INT,
    @Month INT,
    @Year INT
AS
BEGIN
    SET NOCOUNT ON;

    -- Compute the last day of the provided month and year
    DECLARE @LastDayOfMonth DATETIME;
    SET @LastDayOfMonth = EOMONTH(DATEFROMPARTS(@Year, @Month, 1));

    SELECT
        ca.ReferenceNumber,
        MAX(ca.Citizen_id) AS Citizen_id,
        MAX(ca.ServiceId) AS ServiceId,
        MAX(ca.DistrictUidForBank) AS DistrictUidForBank,
        MAX(ca.FormDetails) AS FormDetails,
        MAX(ca.WorkFlow) AS WorkFlow,
        MAX(ca.AdditionalDetails) AS AdditionalDetails,
        MAX(ca.CurrentPlayer) AS CurrentPlayer,
        MAX(ca.[Status]) AS [Status],
        MAX(ca.Created_at) AS Created_at
    FROM
        [dbo].[Citizen_Applications] ca
    CROSS APPLY
        OPENJSON(ca.WorkFlow) AS wf
    OUTER APPLY
        OPENJSON(ca.FormDetails, '$.Location') WITH (
            name NVARCHAR(50) '$.name',
            value INT '$.value'
        ) AS jsonLocation
    OUTER APPLY
        (SELECT
            TRY_CAST(JSON_VALUE(wf.value, '$.completedAt') AS DATETIME) AS CompletedAtDate,
            JSON_VALUE(wf.value, '$.status') AS StatusValue
        ) AS wfParsed
    WHERE
        ca.ServiceId = @ServiceId
        AND (
            (jsonLocation.name = 'District' AND jsonLocation.value = @AccessCode)
            OR (jsonLocation.name = 'Tehsil' AND jsonLocation.value = @AccessCode)
        )
        AND (
            @ApplicationStatus = 'Total Applications'
            OR wfParsed.StatusValue = @ApplicationStatus
        )
        AND wfParsed.StatusValue <> ''
        AND (
            wfParsed.CompletedAtDate IS NULL OR wfParsed.CompletedAtDate <= @LastDayOfMonth
        )
    GROUP BY
        ca.ReferenceNumber;
END;
GO

CREATE PROCEDURE [dbo].[GetRecordsForBankFile_New]
    @AccessCode INT,
    @ApplicationStatus VARCHAR(50) = NULL,
    @ServiceId INT,
    @Month INT,
    @Year INT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @StartOfMonth DATETIME = DATEFROMPARTS(@Year, @Month, 1);
    DECLARE @LastDayOfMonth DATETIME = EOMONTH(@StartOfMonth);
    DECLARE @monthAbbr VARCHAR(3) = UPPER(FORMAT(@StartOfMonth, 'MMM'));
    DECLARE @yearAbbr VARCHAR(2) = RIGHT(CAST(@Year AS VARCHAR), 2);

    DECLARE @districtShort VARCHAR(10);
    DECLARE @division VARCHAR(10);
    DECLARE @bankDetails NVARCHAR(MAX);
    DECLARE @department NVARCHAR(100);

    -- Get District Short Name & Division (Jammu/Kashmir)
    SELECT
        @districtShort = DistrictShort,
        @division = CASE WHEN Division = 1 THEN 'Jammu' WHEN Division = 2 THEN 'Kashmir' END
    FROM District
    WHERE DistrictId = @AccessCode;

    IF @districtShort IS NULL OR @division IS NULL
    BEGIN
        RAISERROR('Invalid District or Division for AccessCode %d.', 16, 1, @AccessCode);
        RETURN;
    END

    -- Get BankDetails JSON + Department Name via DepartmentId
    SELECT
        @bankDetails = s.BankDetails,
        @department = ISNULL(d.DepartmentName, 'Unknown Department')
    FROM Services s
    LEFT JOIN Department d ON s.DepartmentId = d.DepartmentId
    WHERE s.ServiceId = @ServiceId;

    IF @bankDetails IS NULL OR ISJSON(@bankDetails) = 0
    BEGIN
        RAISERROR('Invalid Service or BankDetails JSON for ServiceId %d.', 16, 1, @ServiceId);
        RETURN;
    END

    -- Extract paying bank details for correct division
    DECLARE @payingBankAccountNumber NVARCHAR(50) = JSON_VALUE(@bankDetails, CONCAT('$.', @division, '.AccountNumber'));
    DECLARE @payingBankIfscCode NVARCHAR(20) = JSON_VALUE(@bankDetails, CONCAT('$.', @division, '.IfscCode'));
    DECLARE @payingBankName NVARCHAR(100) = JSON_VALUE(@bankDetails, CONCAT('$.', @division, '.BankName'));

    IF @payingBankAccountNumber IS NULL
        OR @payingBankIfscCode IS NULL
        OR @payingBankName IS NULL
    BEGIN
        RAISERROR('Missing bank details for division %s in ServiceId %d.', 16, 1, @division, @ServiceId);
        RETURN;
    END;

    -- Main query with CTEs
    ;WITH FormDetailsParsed AS (
        SELECT
            ca.ReferenceNumber,
            ca.DistrictUidForBank,
            ca.CurrentPlayer,
            ca.WorkFlow,
            ca.FormDetails,
            MAX(CASE WHEN fields.name = 'ApplicantName' THEN fields.value END) AS applicantName,
            MAX(CASE WHEN fields.name = 'IfscCode' THEN fields.value END) AS receivingIfscCode,
            MAX(CASE WHEN fields.name = 'AccountNumber' THEN fields.value END) AS receivingAccountNumber,
            MAX(CASE WHEN fields.name = 'PensionType' THEN fields.value END) AS pensionType
        FROM Citizen_Applications ca
        CROSS APPLY OPENJSON(ca.FormDetails) AS section
        CROSS APPLY OPENJSON(section.value)
            WITH (
                name NVARCHAR(100) '$.name',
                value NVARCHAR(MAX) '$.value'
            ) fields
        WHERE ca.ServiceId = @ServiceId
          AND ISJSON(ca.FormDetails) = 1
        GROUP BY ca.ReferenceNumber, ca.DistrictUidForBank, ca.CurrentPlayer, ca.WorkFlow, ca.FormDetails
    ),
    WorkflowParsed AS (
        SELECT
            ca.ReferenceNumber,
            JSON_VALUE(ca.WorkFlow, CONCAT('$[', TRY_CAST(ca.CurrentPlayer AS INT), '].status')) AS StatusValue,
            TRY_CAST(JSON_VALUE(ca.WorkFlow, CONCAT('$[', TRY_CAST(ca.CurrentPlayer AS INT), '].completedAt')) AS DATETIME) AS CompletedAtDate
        FROM Citizen_Applications ca
        WHERE ca.ServiceId = @ServiceId
          AND ISJSON(ca.WorkFlow) = 1
          AND TRY_CAST(ca.CurrentPlayer AS INT) IS NOT NULL
    )
    SELECT
        fdp.ReferenceNumber,
        CONCAT(@districtShort, @monthAbbr, @yearAbbr, '00',
               RIGHT('00000000' + ISNULL(CAST(fdp.DistrictUidForBank AS VARCHAR(8)), ''), 8)) AS districtbankuid,
        @department AS department,
        @payingBankAccountNumber AS payingBankAccountNumber,
        @payingBankIfscCode AS payingBankIfscCode,
        @payingBankName AS payingBankName,
        GETDATE() AS fileGenerationDate,
        1000 AS amount,
        fdp.applicantName,
        fdp.receivingIfscCode,
        fdp.receivingAccountNumber,
        fdp.pensionType,
        wp.CompletedAtDate AS sanctionedon
    FROM FormDetailsParsed fdp
    INNER JOIN WorkflowParsed wp ON fdp.ReferenceNumber = wp.ReferenceNumber
    CROSS APPLY OPENJSON(fdp.FormDetails, '$.Location')
        WITH (
            name NVARCHAR(50) '$.name',
            value INT '$.value'
        ) jsonLocation
    WHERE jsonLocation.name = 'District'
      AND jsonLocation.value = @AccessCode
      AND (@ApplicationStatus = 'Total Applications' OR wp.StatusValue = @ApplicationStatus)
      AND wp.StatusValue IS NOT NULL
      AND wp.CompletedAtDate BETWEEN @StartOfMonth AND @LastDayOfMonth;

END;
GO

CREATE PROCEDURE [dbo].[GetShiftedApplications]
    @Role VARCHAR(255),
    @AccessLevel VARCHAR(20), -- Optional now, but keeping for signature consistency
    @AccessCode INT, -- Optional now
    @ServiceId INT
AS
BEGIN
    SET NOCOUNT ON;

    WITH RankedStatus AS (
        SELECT
            ca.ReferenceNumber,
            ca.ReferenceNumberAlphaNumeric,
            ca.Citizen_id,
            ca.ServiceId,
            ca.DistrictUidForBank,
            ca.FormDetails,
            ca.WorkFlow,
            ca.AdditionalDetails,
            ca.CurrentPlayer,
            ca.[Status],
            ca.DataType,
            ca.Created_at,
            jsonWorkFlow.status AS workflow_status,
            jsonWorkFlow.designation,
            jsonWorkFlow.shifted,
            jsonWorkFlow.shiftedFrom,
            ROW_NUMBER() OVER (
                PARTITION BY ca.ReferenceNumber
                ORDER BY
                    CASE WHEN jsonWorkFlow.shifted = 1 THEN 0 ELSE 1 END,
                    jsonWorkFlow.playerId DESC
            ) AS rn
        FROM
            [dbo].[Citizen_Applications] ca
        CROSS APPLY
            OPENJSON(ca.WorkFlow) WITH (
                status NVARCHAR(50) '$.status',
                designation NVARCHAR(50) '$.designation',
                playerId INT '$.playerId',
                shifted BIT '$.shifted',
                shiftedFrom INT '$.shiftedFrom'
            ) AS jsonWorkFlow
        WHERE
            jsonWorkFlow.designation = @Role
            AND ca.ServiceId = @ServiceId
            AND ISJSON(ca.WorkFlow) = 1
            AND jsonWorkFlow.shifted = 1
            AND jsonWorkFlow.shiftedFrom = @AccessCode
    )
    SELECT
        rs.ReferenceNumber,
        MAX(rs.ReferenceNumberAlphaNumeric) AS ReferenceNumberAlphaNumeric,
        MAX(rs.Citizen_id) AS Citizen_id,
        MAX(rs.ServiceId) AS ServiceId,
        MAX(rs.DistrictUidForBank) AS DistrictUidForBank,
        MAX(rs.FormDetails) AS FormDetails,
        MAX(rs.WorkFlow) AS WorkFlow,
        MAX(rs.AdditionalDetails) AS AdditionalDetails,
        MAX(rs.CurrentPlayer) AS CurrentPlayer,
        MAX(rs.[Status]) AS [Status],
        MAX(rs.DataType) AS [DataType],
        MAX(rs.Created_at) AS Created_at
    FROM
        RankedStatus rs
    WHERE
        rs.rn = 1
    GROUP BY
        rs.ReferenceNumber;
END;

CREATE PROCEDURE [dbo].[GetShiftedCount]
    @AccessLevel VARCHAR(20),
    @AccessCode INT,
    @ServiceId INT,
    @TakenBy VARCHAR(50),
    @DivisionCode INT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    WITH RankedStatus AS (
        SELECT
            ca.ReferenceNumber,
            ca.ServiceId,
            ca.FormDetails,
            jsonWorkFlow.status,
            jsonWorkFlow.designation,
            jsonWorkFlow.shifted,
            jsonWorkFlow.shiftedFrom,
            ROW_NUMBER() OVER (
                PARTITION BY ca.ReferenceNumber
                ORDER BY
                    CASE WHEN jsonWorkFlow.shifted = 1 THEN 0 ELSE 1 END, -- Prioritize shifted = true
                    jsonWorkFlow.playerId DESC
            ) AS rn
        FROM
            [dbo].[Citizen_Applications] ca
        CROSS APPLY
            OPENJSON(ca.WorkFlow) WITH (
                status NVARCHAR(50) '$.status',
                designation NVARCHAR(50) '$.designation',
                playerId INT '$.playerId',
                shifted BIT '$.shifted',
                shiftedFrom INT '$.shiftedFrom'
            ) AS jsonWorkFlow
        WHERE
            jsonWorkFlow.designation = @TakenBy
            AND ca.ServiceId = @ServiceId
            AND ISJSON(ca.WorkFlow) = 1
    ),
    LatestStatus AS (
        SELECT
            ReferenceNumber,
            ServiceId,
            FormDetails,
            status,
            shifted,
            shiftedFrom
        FROM
            RankedStatus
        WHERE
            rn = 1
    ),
    FilteredApplications AS (
        SELECT
            ls.ReferenceNumber,
            ls.ServiceId,
            ls.status,
            ls.shifted,
            ls.shiftedFrom
        FROM
            LatestStatus ls
        CROSS APPLY
            OPENJSON(ls.FormDetails, '$.Location') WITH (
                name NVARCHAR(50) '$.name',
                value INT '$.value'
            ) AS jsonLocation
        LEFT JOIN
            [dbo].[District] d ON jsonLocation.name = 'District' AND jsonLocation.value = d.DistrictID
        WHERE
            ls.ServiceId = @ServiceId
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District' AND jsonLocation.name = 'District') -- No comparison with @AccessCode
                OR (@AccessLevel = 'Tehsil' AND jsonLocation.name = 'Tehsil')
                OR (
                    @AccessLevel = 'Division' AND (
                        (jsonLocation.name = 'District' AND d.Division = @DivisionCode)
                        OR
                        (jsonLocation.name = 'Tehsil' AND EXISTS (
                            SELECT 1 FROM Tehsil t
                            INNER JOIN District d2 ON t.DistrictID = d2.DistrictID
                            WHERE t.TehsilID = jsonLocation.value AND d2.Division = @DivisionCode
                        ))
                    )
                )
            )
        GROUP BY ls.ReferenceNumber, ls.ServiceId, ls.status, ls.shifted, ls.shiftedFrom
    )
    SELECT
        ISNULL(SUM(CASE WHEN fa.shifted = 1 AND fa.shiftedFrom = @AccessCode THEN 1 ELSE 0 END), 0) AS ShiftedCount
    FROM
        FilteredApplications fa;
END;

            ) AS rn

        INNER JOIN [dbo].[Citizen_Applications] ca
            ON ca.ReferenceNumber = wa.ReferenceNumber
        CROSS APPLY OPENJSON(wa.WorkFlow) WITH (
            status NVARCHAR(50) '$.status',
            completedAt NVARCHAR(50) '$.completedAt',
            playerId INT '$.playerId',
            designation NVARCHAR(50) '$.designation'
        ) AS jsonWorkFlow
        WHERE
            wa.IsWithheld = 1
            AND wa.ServiceId = @ServiceId
            AND ISJSON(wa.WorkFlow) = 1
            AND ca.DataType != 'legacy'
            AND (@TakenBy IS NULL OR jsonWorkFlow.designation = @TakenBy OR jsonWorkFlow.status = 'returntoedit')
    ),
    LatestWithheld AS (
        SELECT
            ReferenceNumber,
            WithheldType,
            status
        FROM RankedWithheld
        WHERE rn = 1
    ),
    FilteredWithheld AS (
        SELECT DISTINCT
            lw.ReferenceNumber,
            lw.WithheldType,
            lw.status
        FROM LatestWithheld lw
        INNER JOIN [dbo].[Citizen_Applications] ca
            ON ca.ReferenceNumber = lw.ReferenceNumber
        CROSS APPLY OPENJSON(ca.FormDetails, '$.Location') WITH (
            name NVARCHAR(50) '$.name',
            value INT '$.value'
        ) AS jsonLocation
        LEFT JOIN [dbo].[District] d
            ON jsonLocation.name = 'District'
            AND jsonLocation.value = d.DistrictID
        WHERE
            ca.ServiceId = @ServiceId
            AND ISJSON(ca.FormDetails) = 1
            AND ca.DataType != 'legacy'
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District'
                    AND jsonLocation.name = 'District'
                    AND jsonLocation.value = @AccessCode)
                OR (@AccessLevel = 'Tehsil'
                    AND jsonLocation.name = 'Tehsil'
                    AND jsonLocation.value = @AccessCode)
                OR (
                    @AccessLevel = 'Division'
 AND (
                        (jsonLocation.name = 'District'
                            AND d.Division = @DivisionCode)
                        OR
                        (jsonLocation.name = 'Tehsil'
                            AND EXISTS (
                                SELECT 1
                                FROM Tehsil t
                                INNER JOIN District d2
                                    ON t.DistrictID = d2.DistrictID
                                WHERE t.TehsilID = jsonLocation.value
                                    AND d2.Division = @DivisionCode
                            ))
                    )
                )
            )
    )
    -- ======== Final SELECT ========
    SELECT
        -- Main application status counts
        COUNT(DISTINCT CASE WHEN fa.status = 'pending' THEN fa.ReferenceNumber END) AS PendingCount,
        COUNT(DISTINCT CASE WHEN fa.status = 'forwarded' THEN fa.ReferenceNumber END) AS ForwardedCount,
        COUNT(DISTINCT CASE WHEN fa.status = 'returned' THEN fa.ReferenceNumber END) AS ReturnedCount,
        COUNT(DISTINCT CASE WHEN fa.status = 'returntoedit' THEN fa.ReferenceNumber END) AS ReturnToEditCount,
        COUNT(DISTINCT CASE WHEN fa.status = 'sanctioned' THEN fa.ReferenceNumber END) AS SanctionedCount,
        COUNT(DISTINCT CASE WHEN fa.status = 'rejected' THEN fa.ReferenceNumber END) AS RejectCount,
        COUNT(DISTINCT CASE WHEN fa.status = 'disbursed' THEN fa.ReferenceNumber END) AS DisbursedCount,

        -- FIXED TotalApplications
        COUNT(DISTINCT CASE
            WHEN fa.status <> 'returntoedit'
                 OR (fa.status = 'returntoedit' AND fa.designation = @TakenBy)
            THEN fa.ReferenceNumber END
        ) AS TotalApplications,

        -- Forwarded and Sanctioned counts
        ISNULL((SELECT COUNT(*) FROM ForwardedSanctioned), 0) AS ForwardedSanctionedCount,
        ISNULL((SELECT COUNT(*) FROM ForwardedSanctionedCorrigendum), 0) AS ForwardedSanctionedCorrigendumCount,
        ISNULL((SELECT COUNT(*) FROM ForwardedVerifiedCorrection), 0) AS ForwardedVerifiedCorrectionCount,

        -- Corrigendum counts by status
        COUNT(DISTINCT CASE WHEN fc.status = 'pending' AND fc.Type = 'Corrigendum' THEN fc.CorrigendumId END) AS CorrigendumPendingCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'forwarded' AND fc.Type = 'Corrigendum' THEN fc.CorrigendumId END) AS CorrigendumForwardedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'returned' AND fc.Type = 'Corrigendum' THEN fc.CorrigendumId END) AS CorrigendumReturnedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'sanctioned' AND fc.Type = 'Corrigendum' THEN fc.CorrigendumId END) AS CorrigendumSanctionedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'rejected' AND fc.Type = 'Corrigendum' THEN fc.CorrigendumId END) AS CorrigendumRejectedCount,
        COUNT(DISTINCT CASE WHEN fc.Type = 'Corrigendum' THEN fc.CorrigendumId END) AS CorrigendumCount,

        -- Correction counts by status
        COUNT(DISTINCT CASE WHEN fc.status = 'pending' AND fc.Type = 'Correction' THEN fc.CorrigendumId END) AS CorrectionPendingCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'forwarded' AND fc.Type = 'Correction' THEN fc.CorrigendumId END) AS CorrectionForwardedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'returned' AND fc.Type = 'Correction' THEN fc.CorrigendumId END) AS CorrectionReturnedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'verified' AND fc.Type = 'Correction' THEN fc.CorrigendumId END) AS CorrectionSanctionedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'rejected' AND fc.Type = 'Correction' THEN fc.CorrigendumId END) AS CorrectionRejectedCount,
        COUNT(DISTINCT CASE WHEN fc.Type = 'Correction' THEN fc.CorrigendumId END) AS CorrectionCount,

        -- Amendment counts by status
        COUNT(DISTINCT CASE WHEN fc.status = 'pending' AND fc.Type = 'Amendment' THEN fc.CorrigendumId END) AS AmendmentPendingCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'forwarded' AND fc.Type = 'Amendment' THEN fc.CorrigendumId END) AS AmendmentForwardedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'returned' AND fc.Type = 'Amendment' THEN fc.CorrigendumId END) AS AmendmentReturnedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'sanctioned' AND fc.Type = 'Amendment' THEN fc.CorrigendumId END) AS AmendmentSanctionedCount,
        COUNT(DISTINCT CASE WHEN fc.status = 'rejected' AND fc.Type = 'Amendment' THEN fc.CorrigendumId END) AS AmendmentRejectedCount,
        COUNT(DISTINCT CASE WHEN fc.Type = 'Amendment' THEN fc.CorrigendumId END) AS AmendmentCount,

        -- Withheld applications counts
        COUNT(DISTINCT fw.ReferenceNumber) AS TotalWithheldCount,
        COUNT(DISTINCT CASE WHEN fw.WithheldType = 'TEMPORARY' THEN fw.ReferenceNumber END) AS TemporaryWithheldCount,
        COUNT(DISTINCT CASE WHEN fw.WithheldType = 'PERMANENT' THEN fw.ReferenceNumber END) AS PermanentWithheldCount,
        COUNT(DISTINCT CASE WHEN fw.status = 'pending' THEN fw.ReferenceNumber END) AS WithheldPendingCount,
        COUNT(DISTINCT CASE WHEN fw.status = 'forwarded' THEN fw.ReferenceNumber END) AS WithheldForwardedCount,
        COUNT(DISTINCT CASE WHEN fw.status = 'approved' THEN fw.ReferenceNumber END) AS WithheldApprovedCount
    FROM
        FilteredApplications fa
        LEFT JOIN FilteredCorrigendum fc ON fa.ReferenceNumber = fc.ReferenceNumber
        LEFT JOIN FilteredWithheld fw ON fa.ReferenceNumber = fw.ReferenceNumber;
END;

CREATE PROCEDURE [dbo].[GetTemporaryDisabilityCount]
    @AccessLevel VARCHAR(20),
    @AccessCode INT,
    @ServiceId INT,
    @TakenBy VARCHAR(50),
    @DivisionCode INT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- ======== Ranked Workflows Section ========
    WITH RankedStatus AS (
        SELECT
            ca.ReferenceNumber,
            ca.ServiceId,
            ca.FormDetails,
            jsonWorkFlow.status,
            ROW_NUMBER() OVER (
                PARTITION BY ca.ReferenceNumber
                ORDER BY COALESCE(jsonWorkFlow.timestamp, '9999-12-31') DESC, jsonWorkFlow.seq DESC
            ) AS rn
        FROM [dbo].[Citizen_Applications] ca
        CROSS APPLY OPENJSON(ca.WorkFlow) WITH (
            status NVARCHAR(50) '$.status',
            timestamp DATETIME '$.timestamp',
            seq INT '$.seq'
        ) AS jsonWorkFlow
        WHERE
            jsonWorkFlow.status IS NOT NULL
            AND jsonWorkFlow.status <> ''
            AND ca.ServiceId = @ServiceId
            AND ISJSON(ca.WorkFlow) = 1
    ),
    LatestCorrigendum AS (
        SELECT
            c.ReferenceNumber,
            c.CorrigendumFields,
            ROW_NUMBER() OVER (
                PARTITION BY c.ReferenceNumber
                ORDER BY c.CreatedAt DESC
            ) AS rn
        FROM [dbo].[Corrigendum] c
        WHERE
            ISJSON(c.CorrigendumFields) = 1
            AND EXISTS (
                SELECT 1
                FROM OPENJSON(c.CorrigendumFields) WITH (
                    new_value NVARCHAR(100) '$.IfTemporaryDisabilityUdidCardValidUpto.new_value'
                ) AS corr
                WHERE corr.new_value IS NOT NULL
            )
    ),
    LatestStatus AS (
        SELECT
            rs.ReferenceNumber,
            rs.ServiceId,
            rs.FormDetails,
            rs.status,
            lc.CorrigendumFields
        FROM RankedStatus rs
        LEFT JOIN LatestCorrigendum lc
            ON rs.ReferenceNumber = lc.ReferenceNumber
            AND lc.rn = 1
        WHERE rs.rn = 1
    ),
    FilteredApplications AS (
        SELECT
            ls.ReferenceNumber,
            ls.ServiceId,
            ls.FormDetails,
            ls.status,
            ls.CorrigendumFields,
            CASE
                WHEN EXISTS (
                    SELECT 1
                    FROM OPENJSON(ls.FormDetails, '$."Pension Type"') WITH (
                        value NVARCHAR(100) '$.value',
                        additionalFields NVARCHAR(MAX) '$.additionalFields' AS JSON
                    ) AS pension
                    CROSS APPLY OPENJSON(pension.additionalFields) WITH (
                        name NVARCHAR(100) '$.name',
                        value NVARCHAR(100) '$.value',
                        additionalFields NVARCHAR(MAX) '$.additionalFields' AS JSON
                    ) AS disability
                    CROSS APPLY OPENJSON(disability.additionalFields) WITH (
                        name NVARCHAR(100) '$.name',
                        value NVARCHAR(100) '$.value'
                    ) AS tempDisability
                    OUTER APPLY (
                        SELECT new_value
                        FROM OPENJSON(ls.CorrigendumFields) WITH (
                            new_value NVARCHAR(100) '$.IfTemporaryDisabilityUdidCardValidUpto.new_value'
                        )
                    ) AS corr
                    WHERE pension.value = 'PHYSICALLY CHALLENGED PERSON'
                        AND disability.name = 'KindOfDisability'
                        AND disability.value = 'TEMPORARY'
                        AND tempDisability.name = 'IfTemporaryDisabilityUdidCardValidUpto'
                        AND (
                            (corr.new_value IS NOT NULL
                             AND TRY_CONVERT(DATE, corr.new_value) IS NOT NULL
                             AND DATEDIFF(MONTH, GETDATE(), TRY_CONVERT(DATE, corr.new_value)) BETWEEN 0 AND 3)
                            OR
                            (corr.new_value IS NULL
                             AND tempDisability.value IS NOT NULL
                             AND TRY_CONVERT(DATE, tempDisability.value) IS NOT NULL
                             AND DATEDIFF(MONTH, GETDATE(), TRY_CONVERT(DATE, tempDisability.value)) BETWEEN 0 AND 3)
                        )
                ) THEN 1
                ELSE 0
            END AS TemporaryDisabilityExpiringSoon,
            CASE
                WHEN EXISTS (
                    SELECT 1
                    FROM OPENJSON(ls.FormDetails, '$."Pension Type"') WITH (
                        value NVARCHAR(100) '$.value'
                    ) AS pension
                    WHERE pension.value = 'PHYSICALLY CHALLENGED PERSON'
                ) THEN 1
                ELSE 0
            END AS IsPhysicallyChallenged
        FROM LatestStatus ls
        CROSS APPLY OPENJSON(ls.FormDetails, '$.Location') WITH (
            name NVARCHAR(50) '$.name',
            value INT '$.value'
        ) AS jsonLocation
        LEFT JOIN [dbo].[District] d
            ON jsonLocation.name = 'District'
            AND jsonLocation.value = d.DistrictID
        WHERE ls.ServiceId = @ServiceId
            AND (
                @AccessLevel = 'State'
                OR (@AccessLevel = 'District'
                    AND jsonLocation.name = 'District'
                    AND jsonLocation.value = @AccessCode)
                OR (@AccessLevel = 'Tehsil'
                    AND jsonLocation.name = 'Tehsil'
                    AND jsonLocation.value = @AccessCode)
                OR (
                    @AccessLevel = 'Division'
                    AND (
                        (jsonLocation.name = 'District'
                         AND jsonLocation.value = @AccessCode
                         AND d.Division = @DivisionCode)
                        OR
                        (jsonLocation.name = 'Tehsil'
                         AND EXISTS (
                            SELECT 1
                            FROM [dbo].[Tehsil] t
                            INNER JOIN [dbo].[District] d2
                                ON t.DistrictID = d2.DistrictID
                            WHERE t.TehsilID = jsonLocation.value
                                AND d2.Division = @DivisionCode
                         ))
                    )
                )
            )
    )
    -- ======== Final SELECT with Both Counts ========
    SELECT
        ISNULL(SUM(CASE WHEN fa.TemporaryDisabilityExpiringSoon = 1 THEN 1 ELSE 0 END), 0) AS TemporaryDisabilityExpiringSoonCount,
        ISNULL(SUM(CASE WHEN fa.IsPhysicallyChallenged = 1 THEN 1 ELSE 0 END), 0) AS TotalPhysicallyChallengedApplications
    FROM FilteredApplications fa;
END;
GO

CREATE PROCEDURE [dbo].[GetWithheldApplications]
    @ServiceId INT,
    @Type NVARCHAR(50),
    @OfficerRole NVARCHAR(100) = NULL,
    @PageIndex INT = 0,
    @PageSize INT = 10,
    @TotalRecords INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    -- Normalize type
    DECLARE @WithheldType NVARCHAR(50) = CASE
        WHEN @Type LIKE 'withheld_%' THEN RIGHT(@Type, LEN(@Type) - CHARINDEX('_', @Type))
        ELSE @Type
    END;

    DECLARE @IsStatusNotType BIT = CASE
        WHEN @WithheldType NOT IN ('temporary', 'permanent', 'total') THEN 1
        ELSE 0
    END;

    -- Validate OfficerRole when IsStatusNotType = 1
    IF @IsStatusNotType = 1 AND @OfficerRole IS NULL
    BEGIN
        SET @TotalRecords = 0;
        RETURN; -- Exit early with empty result set
    END;

    -- Temporary table to store filtered results
    CREATE TABLE #TempResults (
        Withheld_Id INT,
        ReferenceNumber NVARCHAR(50),
        WithheldType NVARCHAR(20),
        WithheldReason TEXT,
        ApplicantName NVARCHAR(MAX)
    );

    -- Insert filtered data into temporary table
    INSERT INTO #TempResults (Withheld_Id, ReferenceNumber, WithheldType, WithheldReason, ApplicantName)
    SELECT
        wa.Withheld_Id,
        wa.ReferenceNumber,
        wa.WithheldType,
        wa.WithheldReason,
        (
            SELECT TOP 1 value
            FROM OPENJSON(ca.FormDetails, '$."Applicant Details"')
            WITH (
                name NVARCHAR(100) '$.name',
                value NVARCHAR(MAX) '$.value'
            )
            WHERE name = 'ApplicantName'
        ) AS ApplicantName
    FROM [dbo].[Withheld_Applications] wa
    LEFT JOIN [dbo].[Citizen_Applications] ca ON wa.ReferenceNumber = ca.ReferenceNumber
    WHERE wa.ServiceId = @ServiceId
        AND (
            (@IsStatusNotType = 0 AND (@WithheldType = 'total' OR wa.WithheldType = @WithheldType))
            OR
            (@IsStatusNotType = 1 AND EXISTS (
                SELECT 1
                FROM OPENJSON(wa.WorkFlow)
                WITH (
                    status NVARCHAR(50) '$.status',
                    designation NVARCHAR(100) '$.designation'
                ) wf
                WHERE LOWER(wf.status) = LOWER(@WithheldType)
                    AND wf.designation = @OfficerRole
            ))
        );

    -- Set total records in OUTPUT parameter
    SELECT @TotalRecords = COUNT(*) FROM #TempResults;

    -- Return paginated results
    SELECT
        ROW_NUMBER() OVER (ORDER BY Withheld_Id) + (@PageIndex * @PageSize) AS sno,
        ReferenceNumber,
        ISNULL(ApplicantName, 'N/A') AS ApplicantName,
        WithheldType,
        WithheldReason
    FROM #TempResults
    ORDER BY Withheld_Id
    OFFSET @PageIndex * @PageSize ROWS
    FETCH NEXT @PageSize ROWS ONLY;

    -- Clean up
    DROP TABLE #TempResults;
END;
GO

CREATE PROCEDURE [dbo].[ResetUserPassword]
    @Email NVARCHAR(100),
    @UserId INT, -- Adjust type based on your UserId column (e.g., INT, UNIQUEIDENTIFIER, NVARCHAR)
    @NewPassword NVARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;

    -- Declare variables
    DECLARE @HashedPassword VARBINARY(64); -- SHA-256 hash size
    DECLARE @UserExists INT;

    -- Check if the user exists with the given Email and UserId
    SELECT @UserExists = COUNT(*) FROM Users WHERE Email = @Email AND UserId = @UserId;

    IF @UserExists = 0
    BEGIN
        -- Return error if user not found
        SELECT
            Result = 0,
            Message = 'No account found with this email and user ID.',
            UserId = CAST(0 AS INT);
        RETURN;
    END

    -- Hash the new password using SHA-256
    SET @HashedPassword = HASHBYTES('SHA2_256', @NewPassword);

    -- Update the user's password
    UPDATE Users
    SET [Password] = @HashedPassword
    WHERE Email = @Email AND UserId = @UserId;

    -- Return success result with UserId
    SELECT
        Result = 1,
        Message = 'Password reset successfully.',
        UserId = UserId
    FROM Users
    WHERE Email = @Email AND UserId = @UserId;
END;
GO

CREATE PROCEDURE dbo.UpdateAadhaarTokenByReference
    @ReferenceNumber NVARCHAR(100),
    @AadhaarToken NVARCHAR(4000)
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @FormDetails NVARCHAR(MAX);
    DECLARE @NewApplicantJson NVARCHAR(MAX);
    DECLARE @UpdatedFormDetails NVARCHAR(MAX);

    -- 1) Read existing FormDetails
    SELECT @FormDetails = FormDetails
    FROM dbo.Citizen_Applications
    WHERE ReferenceNumber = @ReferenceNumber;

    IF @FormDetails IS NULL OR LTRIM(RTRIM(@FormDetails)) = ''
    BEGIN
        -- initialize empty object so JSON_MODIFY works
        SET @FormDetails = N'{}';
    END

    -- 2) Parse existing Applicant Details elements into a table variable (preserve order)
    DECLARE @app TABLE
    (
        ord INT,
        label NVARCHAR(400),
        name NVARCHAR(400),
        value NVARCHAR(MAX)
    );

    INSERT INTO @app (ord, label, name, value)
    SELECT
        TRY_CAST([key] AS INT) AS ord,
        JSON_VALUE([value], '$.label') AS label,
        JSON_VALUE([value], '$.name') AS name,
        JSON_VALUE([value], '$.value') AS value
    FROM OPENJSON(@FormDetails, '$."Applicant Details"');

    -- 3) If AadhaarNumber exists -> update its value; otherwise append a new object
    IF EXISTS (SELECT 1 FROM @app WHERE LOWER(name) = 'aadhaarnumber')
    BEGIN
        UPDATE @app
        SET value = @AadhaarToken
        WHERE LOWER(name) = 'aadhaarnumber';
    END
    ELSE
    BEGIN
        DECLARE @nextOrd INT = (SELECT ISNULL(MAX(ord), -1) + 1 FROM @app);
        INSERT INTO @app (ord, label, name, value)
        VALUES (@nextOrd, 'Aadhaar Number', 'AadhaarNumber', @AadhaarToken);
    END

    -- 4) Rebuild the Applicant Details JSON array from the table variable
    SELECT @NewApplicantJson = (
        SELECT
            label AS [label],
            name AS [name],
            value AS [value]
        FROM @app
        ORDER BY ord
        FOR JSON PATH
    );

    -- If @NewApplicantJson is NULL (no original Applicant Details and insertion failed),
    -- build a single-entry array
    IF @NewApplicantJson IS NULL
    BEGIN
        SELECT @NewApplicantJson = (
            SELECT 'Aadhaar Number' AS [label], 'AadhaarNumber' AS [name], @AadhaarToken AS [value]
            FOR JSON PATH
        );
    END

    -- 5) Replace or insert the Applicant Details array in FormDetails
    SET @UpdatedFormDetails = JSON_MODIFY(@FormDetails, '$."Applicant Details"', JSON_QUERY(@NewApplicantJson));

    -- 6) Update the table row
    UPDATE dbo.Citizen_Applications
    SET FormDetails = @UpdatedFormDetails
    WHERE ReferenceNumber = @ReferenceNumber;

    -- No output required per request
END;
GO

            FOR JSON PATH
        ))
    )
    FROM Citizen_Applications ca

CREATE PROCEDURE [dbo].[ValidateIFSC]
    @bankName VARCHAR(255),
    @ifscCode VARCHAR(11)
AS
BEGIN
    SET NOCOUNT ON;

    -- Select only the record where IFSC matches exactly, but only if the first 5 characters also match
    SELECT *
    FROM BankDetails
    WHERE BANK LIKE '%' + @bankName + '%'
      AND IFSC = @ifscCode
      AND LEFT(IFSC, 5) = LEFT(@ifscCode, 5);
END;
GO
